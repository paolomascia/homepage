Testing & Benchmarking in Rust

Rust Testing Benchmarking

1. Unit Tests

Rust unit tests live alongside the code they test in a #[cfg(test)] module. Run them with cargo test.

// src/lib.rs

pub fn add(a: i32, b: i32) -> i32 { a + b }

pub fn divide(a: f64, b: f64) -> Result<f64, String> {
    if b == 0.0 {
        Err("division by zero".to_string())
    } else {
        Ok(a / b)
    }
}

pub fn is_palindrome(s: &str) -> bool {
    let chars: Vec<char> = s.chars().collect();
    chars == chars.iter().rev().cloned().collect::<Vec<_>>()
}

#[cfg(test)]      // only compiled during `cargo test`
mod tests {
    use super::*; // import everything from the parent module

    #[test]
    fn test_add() {
        assert_eq!(add(2, 3), 5);
        assert_eq!(add(-1, 1), 0);
        assert_eq!(add(0, 0), 0);
    }

    #[test]
    fn test_divide_ok() {
        let result = divide(10.0, 2.0).unwrap();
        assert!((result - 5.0).abs() < f64::EPSILON);
    }

    #[test]
    fn test_divide_by_zero() {
        let result = divide(1.0, 0.0);
        assert!(result.is_err());
        assert_eq!(result.unwrap_err(), "division by zero");
    }

    #[test]
    #[should_panic(expected = "attempt to divide by zero")]
    fn test_integer_divide_by_zero() {
        let _ = 5 / 0; // panics with "attempt to divide by zero"
    }

    #[test]
    fn test_palindrome() {
        assert!(is_palindrome("racecar"));
        assert!(is_palindrome(""));
        assert!(!is_palindrome("hello"));
    }

    // Test that returns Result — failure is an Err, not a panic
    #[test]
    fn test_divide_returns_result() -> Result<(), String> {
        let x = divide(9.0, 3.0)?;
        assert_eq!(x, 3.0);
        Ok(())
    }
}

Common assert macros:

  • assert!(expr) — panics if false
  • assert_eq!(a, b) — panics if a != b, shows both values
  • assert_ne!(a, b) — panics if a == b
  • assert!(expr, "message {}", val) — custom message with formatting

2. Test Organization and Filtering

// Group related tests in sub-modules
#[cfg(test)]
mod tests {
    mod arithmetic {
        use super::super::*;

        #[test]
        fn add_positive() { assert_eq!(add(1, 2), 3); }

        #[test]
        fn add_negative() { assert_eq!(add(-5, 3), -2); }
    }

    mod strings {
        use super::super::*;

        #[test]
        fn palindrome_even() { assert!(is_palindrome("abba")); }

        #[test]
        fn palindrome_odd() { assert!(is_palindrome("aba")); }
    }
}

# Run all tests
cargo test

# Run tests matching a name pattern
cargo test palindrome       # runs tests whose name contains "palindrome"
cargo test tests::strings   # runs all tests in the strings module

# Run a single test by exact name
cargo test tests::arithmetic::add_positive -- --exact

# Show stdout even for passing tests
cargo test -- --nocapture

# Run tests sequentially (not in parallel)
cargo test -- --test-threads=1

# List all test names without running them
cargo test -- --list

    }

    // Conditional compilation: only run on Linux

}

3. Integration Tests

Integration tests live in the tests/ directory. Each file is compiled as a separate crate that links against your library — they test the public API only.

// tests/api.rs — tests the public API of the library crate
use my_crate::{add, divide, is_palindrome};

#[test]
fn api_add_works() {
    assert_eq!(add(10, 20), 30);
}

#[test]
fn api_divide_by_zero_returns_err() {
    assert!(divide(1.0, 0.0).is_err());
}

// tests/common/mod.rs — shared helpers for multiple test files
pub fn assert_approx_eq(a: f64, b: f64) {
    assert!((a - b).abs() < 1e-9, "expected {a} ≈ {b}");
}

pub fn make_test_data() -> Vec<i32> {
    (1..=10).collect()
}

// tests/integration.rs — uses the shared helpers
mod common;

#[test]
fn test_with_helpers() {
    let data = common::make_test_data();
    assert_eq!(data.len(), 10);
    common::assert_approx_eq(1.0 / 3.0 * 3.0, 1.0);
}

# Run only integration tests
cargo test --test api
cargo test --test integration

4. Documentation Tests

Code blocks in doc comments are compiled and run as tests. This ensures documentation examples stay correct as the code evolves.

/// Computes the factorial of `n`.
///
/// # Examples
///
/// ```
/// use my_crate::factorial;
///
/// assert_eq!(factorial(0), 1);
/// assert_eq!(factorial(5), 120);
/// assert_eq!(factorial(10), 3_628_800);
/// ```
///
/// # Panics
///
/// Panics if `n` would overflow `u64`.
///
/// ```should_panic
/// use my_crate::factorial;
/// factorial(21); // overflows u64
/// ```
pub fn factorial(n: u64) -> u64 {
    (1..=n).product()
}

/// Returns the nth Fibonacci number.
///
/// ```
/// use my_crate::fib;
/// let seq: Vec<u64> = (0..8).map(fib).collect();
/// assert_eq!(seq, [0, 1, 1, 2, 3, 5, 8, 13]);
/// ```
pub fn fib(n: u64) -> u64 {
    match n {
        0 => 0,
        1 => 1,
        _ => fib(n - 1) + fib(n - 2),
    }
}

4.1 Doc Test Annotations

/// ```ignore
/// // This block is in the docs but NOT compiled as a test
/// some_unstable_api();
/// ```
///
/// ```no_run
/// // This block IS compiled but NOT executed (e.g., would block or need network)
/// let listener = std::net::TcpListener::bind("0.0.0.0:80").unwrap();
/// ```
///
/// ```compile_fail
/// // This block must FAIL to compile — good for showing type errors
/// let x: i32 = "not a number"; // error[E0308]
/// ```
///
/// ```
/// # // Lines starting with `#` are compiled but hidden in docs
/// # use my_crate::add;
/// assert_eq!(add(2, 3), 5); // only this line shows in rendered docs
/// ```
pub fn documented_function() {}

# Run only documentation tests
cargo test --doc

5. Async Tests

// Cargo.toml: tokio = { version = "1", features = ["full", "test-util"] }

#[cfg(test)]
mod async_tests {
    use tokio::time::{sleep, Duration};

    // #[tokio::test] sets up a Tokio runtime for the test
    #[tokio::test]
    async fn test_async_operation() {
        sleep(Duration::from_millis(10)).await;
        assert_eq!(2 + 2, 4);
    }

    // Single-threaded runtime for tests without Send requirements
    #[tokio::test(flavor = "current_thread")]
    async fn test_single_thread() {
        let result = async { 42u32 }.await;
        assert_eq!(result, 42);
    }

    // tokio::test with time control
    #[tokio::test]
    async fn test_with_pause() {
        // tokio::time::pause() stops the mock clock — sleep(1h) returns instantly
        tokio::time::pause();
        let start = tokio::time::Instant::now();
        sleep(Duration::from_secs(3600)).await; // returns immediately!
        assert!(start.elapsed() > Duration::from_secs(3600));
    }

    // Test with timeout
    #[tokio::test]
    async fn test_with_timeout() {
        let result = tokio::time::timeout(
            Duration::from_millis(100),
            async { sleep(Duration::from_millis(10)).await; 42u32 },
        )
        .await;
        assert_eq!(result.unwrap(), 42);
    }
}

6. Mocking and Test Doubles

Rust's trait system makes mocking straightforward without a framework — implement a test double that records calls or returns fixed values.

// Define behavior through a trait
trait EmailSender: Send + Sync {
    fn send(&self, to: &str, subject: &str, body: &str) -> Result<(), String>;
}

// Production implementation
struct SmtpSender { host: String }
impl EmailSender for SmtpSender {
    fn send(&self, to: &str, subject: &str, body: &str) -> Result<(), String> {
        // real SMTP logic
        println!("[SMTP] to={to} subject={subject}");
        Ok(())
    }
}

// The code under test
struct UserService {
    sender: Box<dyn EmailSender>,
}

impl UserService {
    fn welcome_user(&self, email: &str) -> Result<(), String> {
        self.sender.send(email, "Welcome!", "Thanks for signing up.")
    }
}

#[cfg(test)]
mod tests {
    use super::*;
    use std::sync::{Arc, Mutex};

    // Test double that records sent emails
    struct MockSender {
        sent: Arc<Mutex<Vec<(String, String, String)>>>,
        should_fail: bool,
    }

    impl MockSender {
        fn new() -> Self {
            MockSender { sent: Arc::new(Mutex::new(Vec::new())), should_fail: false }
        }
        fn failing() -> Self {
            MockSender { sent: Arc::new(Mutex::new(Vec::new())), should_fail: true }
        }
    }

    impl EmailSender for MockSender {
        fn send(&self, to: &str, subject: &str, body: &str) -> Result<(), String> {
            if self.should_fail {
                return Err("SMTP unavailable".to_string());
            }
            self.sent.lock().unwrap()
                .push((to.to_string(), subject.to_string(), body.to_string()));
            Ok(())
        }
    }

    #[test]
    fn test_welcome_sends_email() {
        let mock = MockSender::new();
        let sent = Arc::clone(&mock.sent);
        let service = UserService { sender: Box::new(mock) };

        service.welcome_user("alice@example.com").unwrap();

        let emails = sent.lock().unwrap();
        assert_eq!(emails.len(), 1);
        assert_eq!(emails[0].0, "alice@example.com");
        assert_eq!(emails[0].1, "Welcome!");
    }

    #[test]
    fn test_welcome_propagates_error() {
        let service = UserService { sender: Box::new(MockSender::failing()) };
        let result = service.welcome_user("bob@example.com");
        assert!(result.is_err());
        assert!(result.unwrap_err().contains("SMTP unavailable"));
    }
}

6.1 mockall Crate

// Cargo.toml: mockall = "0.13"
// mockall auto-generates mock structs from traits
use mockall::{automock, predicate::*};

#[automock]
trait Database {
    fn find_user(&self, id: u32) -> Option<String>;
    fn save_user(&mut self, id: u32, name: &str) -> bool;
}

#[cfg(test)]
mod mock_tests {
    use super::*;

    #[test]
    fn test_with_mockall() {
        let mut db = MockDatabase::new();

        // Set expectations
        db.expect_find_user()
            .with(eq(42))
            .times(1)
            .returning(|_| Some("Alice".to_string()));

        db.expect_find_user()
            .with(eq(0))
            .returning(|_| None);

        assert_eq!(db.find_user(42), Some("Alice".to_string()));
        assert_eq!(db.find_user(0), None);
        // mockall verifies expectations were met when `db` is dropped
    }
}

7. Property-Based Testing with proptest

proptest generates hundreds of random inputs and checks that properties hold for all of them. It also shrinks failing inputs to the minimal example.

// Cargo.toml: proptest = "1"
use proptest::prelude::*;

fn sort(mut v: Vec<i32>) -> Vec<i32> {
    v.sort();
    v
}

proptest! {
    // Automatically generates random Vec<i32> inputs
    #[test]
    fn sort_is_idempotent(v: Vec<i32>) {
        let once = sort(v.clone());
        let twice = sort(once.clone());
        prop_assert_eq!(once, twice);
    }

    #[test]
    fn sort_preserves_length(v: Vec<i32>) {
        prop_assert_eq!(sort(v.clone()).len(), v.len());
    }

    #[test]
    fn sort_is_ordered(v: Vec<i32>) {
        let sorted = sort(v);
        for w in sorted.windows(2) {
            prop_assert!(w[0] <= w[1]);
        }
    }

    // Custom strategy: only test with non-empty vecs of small numbers
    #[test]
    fn sort_min_is_first(v in prop::collection::vec(0i32..100, 1..50)) {
        let sorted = sort(v.clone());
        let min = *v.iter().min().unwrap();
        prop_assert_eq!(sorted[0], min);
    }
}

7.1 Custom Strategies

use proptest::prelude::*;

// Generate a valid email address (simplified)
fn email_strategy() -> impl Strategy<Value = String> {
    ("[a-z]{3,10}", "[a-z]{3,8}").prop_map(|(user, domain)| {
        format!("{user}@{domain}.com")
    })
}

// Generate a pair where a <= b
fn ordered_pair() -> impl Strategy<Value = (i32, i32)> {
    (i32::MIN / 2..i32::MAX / 2).prop_flat_map(|a| {
        (Just(a), a..i32::MAX / 2)
    })
}

proptest! {
    #[test]
    fn email_contains_at(email in email_strategy()) {
        prop_assert!(email.contains('@'));
    }

    #[test]
    fn pair_is_ordered((a, b) in ordered_pair()) {
        prop_assert!(a <= b);
    }
}

8. Benchmarking with Criterion

Criterion provides statistically rigorous benchmarks with warm-up, outlier detection, and HTML reports.

# Cargo.toml
[dev-dependencies]
criterion = { version = "0.5", features = ["html_reports"] }

[[bench]]
name = "my_benchmark"
harness = false   # disable the default test harness

// benches/my_benchmark.rs
use criterion::{black_box, criterion_group, criterion_main, Criterion, BenchmarkId};

fn fibonacci(n: u64) -> u64 {
    match n { 0 => 0, 1 => 1, n => fibonacci(n-1) + fibonacci(n-2) }
}

fn fibonacci_iter(n: u64) -> u64 {
    let (mut a, mut b) = (0u64, 1u64);
    for _ in 0..n { (a, b) = (b, a + b); }
    a
}

// Single benchmark
fn bench_fibonacci(c: &mut Criterion) {
    c.bench_function("fib 20 recursive", |b| {
        // black_box prevents the compiler from optimizing away the computation
        b.iter(|| fibonacci(black_box(20)))
    });
}

// Parametric benchmark — compare across different inputs
fn bench_fib_compare(c: &mut Criterion) {
    let mut group = c.benchmark_group("fibonacci");

    for n in [10u64, 20, 30].iter() {
        group.bench_with_input(BenchmarkId::new("recursive", n), n, |b, &n| {
            b.iter(|| fibonacci(black_box(n)))
        });
        group.bench_with_input(BenchmarkId::new("iterative", n), n, |b, &n| {
            b.iter(|| fibonacci_iter(black_box(n)))
        });
    }

    group.finish();
}

// Sort benchmark with setup
fn bench_sort(c: &mut Criterion) {
    let mut group = c.benchmark_group("sort");

    for size in [100usize, 1_000, 10_000] {
        group.bench_with_input(
            BenchmarkId::new("std_sort", size),
            &size,
            |b, &size| {
                // iter_batched: re-creates the input for each iteration
                b.iter_batched(
                    || (0..size as i32).rev().collect::<Vec<i32>>(),
                    |mut v| { v.sort(); v },
                    criterion::BatchSize::SmallInput,
                )
            }
        );
    }

    group.finish();
}

criterion_group!(benches, bench_fibonacci, bench_fib_compare, bench_sort);
criterion_main!(benches);

# Run benchmarks
cargo bench

# Run a specific benchmark
cargo bench -- fibonacci

# Save a baseline and compare later
cargo bench -- --save-baseline main
git checkout feature-branch
cargo bench -- --baseline main   # shows regression/improvement vs main

# HTML reports are at target/criterion/report/index.html

10. CI Configuration Example

# .github/workflows/test.yml
name: Tests

on: [push, pull_request]

env:
  CARGO_TERM_COLOR: always
  RUST_BACKTRACE: 1

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-toolchain@stable
      - uses: Swatinem/rust-cache@v2

      # Build
      - run: cargo build --all-features

      # Unit and integration tests
      - run: cargo test --all-features

      # Documentation tests
      - run: cargo test --doc

      # Clippy lints
      - run: cargo clippy -- -D warnings

      # Format check
      - run: cargo fmt --check

  coverage:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-toolchain@stable
      - uses: taiki-e/install-action@cargo-llvm-cov
      - run: cargo llvm-cov --lcov --output-path lcov.info
      - uses: codecov/codecov-action@v4
        with:
          files: lcov.info
