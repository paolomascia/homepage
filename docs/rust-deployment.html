Deploying Rust in Production

Rust Docker Production

1. Docker Multi-Stage Builds

Multi-stage builds separate compilation from the final runtime image, producing minimal containers with no Rust toolchain overhead.

# Dockerfile — multi-stage Rust build

# ── Stage 1: dependency cache ─────────────────────────────────────────
# Caching dependencies separately avoids rebuilding on every source change
FROM rust:1.77-slim AS chef
WORKDIR /app
RUN cargo install cargo-chef

FROM chef AS planner
COPY . .
RUN cargo chef prepare --recipe-path recipe.json

FROM chef AS builder
COPY --from=planner /app/recipe.json recipe.json
# Build only dependencies (cached unless Cargo.toml changes)
RUN cargo chef cook --release --recipe-path recipe.json

# ── Stage 2: compile the application ──────────────────────────────────
COPY . .
RUN cargo build --release --bin myapp

# ── Stage 3: minimal runtime image ────────────────────────────────────
# distroless/cc-debian12 has glibc + libssl but no shell
FROM gcr.io/distroless/cc-debian12 AS runtime
WORKDIR /app

# Copy the compiled binary
COPY --from=builder /app/target/release/myapp /app/myapp

# Non-root user for security (distroless provides 'nonroot')
USER nonroot

EXPOSE 8080
ENTRYPOINT ["/app/myapp"]

# Alternative: musl static binary — truly zero-dependency
FROM rust:1.77 AS builder
WORKDIR /app

# Add musl target
RUN rustup target add x86_64-unknown-linux-musl
RUN apt-get update && apt-get install -y musl-tools

COPY Cargo.toml Cargo.lock ./
COPY src ./src

RUN cargo build --release --target x86_64-unknown-linux-musl

# Use scratch (empty image) — possible only with static binaries
FROM scratch
COPY --from=builder /app/target/x86_64-unknown-linux-musl/release/myapp /myapp
EXPOSE 8080
ENTRYPOINT ["/myapp"]

# Build and size comparison
docker build -t myapp:latest .
docker images myapp

# REPOSITORY   TAG      IMAGE ID       SIZE
# myapp        latest   abc123def456   12.4MB   ← vs ~1.5GB for rust:latest

# Multi-platform build for ARM64 (Apple Silicon / AWS Graviton)
docker buildx build --platform linux/amd64,linux/arm64 -t myapp:latest --push .

2. Graceful Shutdown

Production services must drain in-flight requests before exiting on SIGTERM. Kubernetes sends SIGTERM before SIGKILL with a default 30-second grace period.

// src/main.rs — graceful shutdown with tokio
use axum::{routing::get, Router};
use tokio::net::TcpListener;
use tokio::signal;
use std::net::SocketAddr;

#[tokio::main]
async fn main() -> anyhow::Result<()> {
    let app = Router::new()
        .route("/", get(|| async { "Hello" }))
        .route("/health", get(health_handler));

    let addr = SocketAddr::from(([0, 0, 0, 0], 8080));
    let listener = TcpListener::bind(addr).await?;

    tracing::info!("listening on {addr}");

    axum::serve(listener, app)
        .with_graceful_shutdown(shutdown_signal())
        .await?;

    tracing::info!("server shut down cleanly");
    Ok(())
}

// Waits for either Ctrl-C or SIGTERM
async fn shutdown_signal() {
    let ctrl_c = async {
        signal::ctrl_c().await.expect("failed to install Ctrl-C handler");
    };

    #[cfg(unix)]
    let terminate = async {
        signal::unix::signal(signal::unix::SignalKind::terminate())
            .expect("failed to install SIGTERM handler")
            .recv()
            .await;
    };

    #[cfg(not(unix))]
    let terminate = std::future::pending::<()>();

    tokio::select! {
        _ = ctrl_c   => tracing::info!("received Ctrl-C"),
        _ = terminate => tracing::info!("received SIGTERM"),
    }

    tracing::info!("initiating graceful shutdown…");
    // Axum will stop accepting new connections and wait for existing ones to finish
}

async fn health_handler() -> &'static str { "OK" }

// More complete shutdown: notify background tasks via CancellationToken
use tokio_util::sync::CancellationToken;
use tokio::sync::mpsc;

#[tokio::main]
async fn main() -> anyhow::Result<()> {
    let token = CancellationToken::new();

    // Spawn background tasks that respect cancellation
    let bg_token = token.clone();
    let bg = tokio::spawn(async move {
        let mut interval = tokio::time::interval(std::time::Duration::from_secs(5));
        loop {
            tokio::select! {
                _ = interval.tick() => {
                    tracing::info!("background tick");
                    // e.g. flush metrics, evict cache entries
                }
                _ = bg_token.cancelled() => {
                    tracing::info!("background task shutting down");
                    break;
                }
            }
        }
    });

    // HTTP server shuts down on signal
    let server_token = token.clone();
    let server = tokio::spawn(async move {
        let app = Router::new().route("/health", get(health_handler));
        let listener = TcpListener::bind("0.0.0.0:8080").await.unwrap();
        axum::serve(listener, app)
            .with_graceful_shutdown(async move { server_token.cancelled().await })
            .await
            .unwrap();
    });

    // Wait for OS signal, then cancel everything
    shutdown_signal().await;
    token.cancel();

    // Wait for all tasks to finish
    let _ = tokio::join!(server, bg);
    tracing::info!("all tasks finished, exiting");
    Ok(())
}

3. Health Checks and Readiness Probes

Kubernetes uses liveness and readiness probes to manage traffic routing and pod restarts:

use axum::{routing::get, Router, Json, extract::State, http::StatusCode};
use serde::Serialize;
use std::sync::Arc;
use tokio::sync::RwLock;

#[derive(Debug, Clone)]
struct AppState {
    db_pool: sqlx::PgPool,
    ready: Arc<RwLock<bool>>,
}

#[derive(Serialize)]
struct HealthResponse {
    status: &'static str,
    version: &'static str,
    checks: std::collections::HashMap<String, CheckResult>,
}

#[derive(Serialize)]
struct CheckResult {
    ok: bool,
    #[serde(skip_serializing_if = "Option::is_none")]
    error: Option<String>,
}

// /health/live — is the process alive? (never depends on external systems)
async fn liveness() -> StatusCode {
    StatusCode::OK
}

// /health/ready — can we serve traffic? (checks DB, cache, etc.)
async fn readiness(State(state): State<AppState>) -> (StatusCode, Json<HealthResponse>) {
    let mut checks = std::collections::HashMap::new();
    let mut all_ok = true;

    // Check database connectivity
    let db_ok = sqlx::query("SELECT 1")
        .execute(&state.db_pool)
        .await
        .is_ok();
    if !db_ok { all_ok = false; }
    checks.insert("database".into(), CheckResult { ok: db_ok, error: if db_ok { None } else { Some("connection failed".into()) } });

    // Check application readiness flag (set after startup tasks complete)
    let app_ready = *state.ready.read().await;
    if !app_ready { all_ok = false; }
    checks.insert("startup".into(), CheckResult { ok: app_ready, error: if app_ready { None } else { Some("migrations not complete".into()) } });

    let status = if all_ok { StatusCode::OK } else { StatusCode::SERVICE_UNAVAILABLE };
    (status, Json(HealthResponse {
        status: if all_ok { "healthy" } else { "degraded" },
        version: env!("CARGO_PKG_VERSION"),
        checks,
    }))
}

fn health_router(state: AppState) -> Router {
    Router::new()
        .route("/health/live", get(liveness))
        .route("/health/ready", get(readiness))
        .with_state(state)
}

4. Configuration Management

Use the config crate to layer configuration from files, environment variables, and command-line arguments:

# Cargo.toml
[dependencies]
config  = "0.14"
serde   = { version = "1", features = ["derive"] }
dotenvy = "0.15"      # load .env files in development

use config::{Config, ConfigError, Environment, File};
use serde::Deserialize;

#[derive(Debug, Deserialize, Clone)]
pub struct Settings {
    pub server: ServerSettings,
    pub database: DatabaseSettings,
    pub redis: RedisSettings,
    pub jwt: JwtSettings,
}

#[derive(Debug, Deserialize, Clone)]
pub struct ServerSettings {
    pub host: String,
    pub port: u16,
    pub workers: usize,
    pub timeout_secs: u64,
}

#[derive(Debug, Deserialize, Clone)]
pub struct DatabaseSettings {
    pub url: String,         // can be set via DATABASE_URL env var
    pub max_connections: u32,
    pub min_connections: u32,
}

#[derive(Debug, Deserialize, Clone)]
pub struct RedisSettings {
    pub url: String,
    pub pool_size: u32,
}

#[derive(Debug, Deserialize, Clone)]
pub struct JwtSettings {
    pub secret: String,     // loaded from secret mount or env
    pub expiry_secs: u64,
}

impl Settings {
    pub fn load() -> Result<Self, ConfigError> {
        let env = std::env::var("APP_ENV").unwrap_or_else(|_| "development".into());

        Config::builder()
            // 1. Base defaults
            .add_source(File::with_name("config/default"))
            // 2. Environment-specific overrides (optional)
            .add_source(File::with_name(&format!("config/{env}")).required(false))
            // 3. Local overrides — never committed to git
            .add_source(File::with_name("config/local").required(false))
            // 4. Environment variables with APP__ prefix
            //    e.g. APP__DATABASE__URL overrides settings.database.url
            .add_source(
                Environment::with_prefix("APP")
                    .prefix_separator("__")
                    .separator("__")
            )
            .build()?
            .try_deserialize()
    }
}

5. Secrets Management

// Load secrets from environment at startup — fail fast if missing
use std::env;

pub struct Secrets {
    pub jwt_secret: String,
    pub database_url: String,
    pub api_key: String,
}

impl Secrets {
    pub fn from_env() -> anyhow::Result<Self> {
        Ok(Self {
            jwt_secret: env::var("JWT_SECRET")
                .map_err(|_| anyhow::anyhow!("JWT_SECRET is required"))?,
            database_url: env::var("DATABASE_URL")
                .map_err(|_| anyhow::anyhow!("DATABASE_URL is required"))?,
            api_key: env::var("EXTERNAL_API_KEY")
                .map_err(|_| anyhow::anyhow!("EXTERNAL_API_KEY is required"))?,
        })
    }
}

// Prevent secrets from appearing in Debug output or logs
use std::fmt;

pub struct Secret(String);

impl Secret {
    pub fn new(value: String) -> Self { Self(value) }
    pub fn expose(&self) -> &str { &self.0 }
}

impl fmt::Debug for Secret {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        f.write_str("[REDACTED]")
    }
}

impl fmt::Display for Secret {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        f.write_str("[REDACTED]")
    }
}

// Usage
#[derive(Debug)]
struct AppConfig {
    host: String,
    jwt_secret: Secret,  // won't appear in logs
}

// AWS Secrets Manager integration (aws-sdk-secretsmanager)
use aws_config::BehaviorVersion;
use aws_sdk_secretsmanager::Client;

async fn load_from_aws_secrets(secret_name: &str) -> anyhow::Result<String> {
    let config = aws_config::load_defaults(BehaviorVersion::latest()).await;
    let client = Client::new(&config);

    let response = client
        .get_secret_value()
        .secret_id(secret_name)
        .send()
        .await?;

    response.secret_string()
        .map(|s| s.to_string())
        .ok_or_else(|| anyhow::anyhow!("secret has no string value"))
}

// Cache secrets with a TTL to avoid hitting the API on every request
use std::sync::Arc;
use tokio::sync::RwLock;
use std::time::{Duration, Instant};

struct CachedSecret {
    value: String,
    fetched_at: Instant,
    ttl: Duration,
}

impl CachedSecret {
    fn is_fresh(&self) -> bool { self.fetched_at.elapsed() < self.ttl }
}

6. Environment-Aware Logging

use tracing_subscriber::{layer::SubscriberExt, util::SubscriberInitExt, EnvFilter};

pub fn init_tracing(env: &str) {
    let filter = EnvFilter::try_from_default_env()
        .unwrap_or_else(|_| EnvFilter::new(if env == "production" { "info" } else { "debug" }));

    let registry = tracing_subscriber::registry().with(filter);

    if env == "production" {
        // Machine-readable JSON in production (for log aggregators like Datadog, Loki)
        let json_layer = tracing_subscriber::fmt::layer()
            .json()
            .with_current_span(true)
            .with_span_list(true)
            .with_target(true);
        registry.with(json_layer).init();
    } else {
        // Human-readable in development
        let pretty_layer = tracing_subscriber::fmt::layer()
            .pretty()
            .with_file(true)
            .with_line_number(true);
        registry.with(pretty_layer).init();
    }
}

fn main() {
    let env = std::env::var("APP_ENV").unwrap_or_else(|_| "development".into());
    init_tracing(&env);

    tracing::info!(env = %env, version = env!("CARGO_PKG_VERSION"), "starting application");
}

7. Release Builds and Binary Optimization

# Cargo.toml — production profile
[profile.release]
opt-level     = 3          # maximum optimization
lto           = "fat"      # link-time optimization across crates
codegen-units = 1          # single codegen unit (slower compile, faster binary)
panic         = "abort"    # no stack unwinding on panic (smaller binary)
strip         = "symbols"  # strip debug symbols from final binary
debug         = false

# PGO: profile-guided optimization (optional, significant gains for CPU-bound code)
# 1. Build with instrumentation: RUSTFLAGS="-Cprofile-generate=/tmp/pgo-data" cargo build --release
# 2. Run representative workload against instrumented binary
# 3. Merge: llvm-profdata merge -output=/tmp/merged.profdata /tmp/pgo-data
# 4. Rebuild: RUSTFLAGS="-Cprofile-use=/tmp/merged.profdata" cargo build --release

# Embed build metadata into the binary at compile time
# build.rs
fn main() {
    // Make GIT_COMMIT available as env!("GIT_COMMIT") at compile time
    let output = std::process::Command::new("git")
        .args(["rev-parse", "--short", "HEAD"])
        .output()
        .unwrap_or_default();
    let commit = String::from_utf8(output.stdout).unwrap_or_default();
    println!("cargo:rustc-env=GIT_COMMIT={}", commit.trim());

    // BUILD_DATE
    println!("cargo:rustc-env=BUILD_DATE={}", chrono::Utc::now().format("%Y-%m-%d"));

    println!("cargo:rerun-if-changed=.git/HEAD");
}

// Expose version info in the health endpoint
#[derive(serde::Serialize)]
struct BuildInfo {

async fn version_handler() -> axum::Json<BuildInfo> {
    axum::Json(BuildInfo {

8. CI/CD Pipeline

# .github/workflows/release.yml
name: Build & Release

on:
  push:
    tags: ["v*.*.*"]

env:
  REGISTRY: ghcr.io
  IMAGE_NAME: ${{ github.repository }}

jobs:
  # ── Run tests before building the image ─────────────────────────────
  test:
    runs-on: ubuntu-latest
    services:
      postgres:
        image: postgres:16
        env:
          POSTGRES_PASSWORD: postgres
          POSTGRES_DB: myapp_test
        options: >-
          --health-cmd pg_isready
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5
        ports:
          - 5432:5432
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-toolchain@stable
      - uses: Swatinem/rust-cache@v2
      - name: Run tests
        run: cargo test --all-features
        env:
          DATABASE_URL: postgres://postgres:postgres@localhost/myapp_test

  # ── Build multi-platform Docker image ───────────────────────────────
  docker:
    needs: test
    runs-on: ubuntu-latest
    permissions:
      contents: read
      packages: write
    steps:
      - uses: actions/checkout@v4

      - name: Set up QEMU (for ARM64 cross-compilation)
        uses: docker/setup-qemu-action@v3

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: Log in to GHCR
        uses: docker/login-action@v3
        with:
          registry: ${{ env.REGISTRY }}
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Extract metadata
        id: meta
        uses: docker/metadata-action@v5
        with:
          images: ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}
          tags: |
            type=semver,pattern={{version}}
            type=semver,pattern={{major}}.{{minor}}
            type=sha,prefix=sha-

      - name: Build and push
        uses: docker/build-push-action@v5
        with:
          context: .
          platforms: linux/amd64,linux/arm64
          push: true
          tags: ${{ steps.meta.outputs.tags }}
          labels: ${{ steps.meta.outputs.labels }}
          cache-from: type=gha
          cache-to: type=gha,mode=max

  # ── Build static binaries for multiple platforms ─────────────────────
  binaries:
    needs: test
    strategy:
      matrix:
        include:
          - target: x86_64-unknown-linux-musl
            os: ubuntu-latest
          - target: aarch64-unknown-linux-musl
            os: ubuntu-latest
          - target: x86_64-apple-darwin
            os: macos-latest
          - target: aarch64-apple-darwin
            os: macos-latest
          - target: x86_64-pc-windows-msvc
            os: windows-latest
    runs-on: ${{ matrix.os }}
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-toolchain@stable
        with:
          targets: ${{ matrix.target }}
      - uses: Swatinem/rust-cache@v2
      - name: Install musl tools
        if: contains(matrix.target, 'musl')
        run: sudo apt-get install -y musl-tools
      - name: Build
        run: cargo build --release --target ${{ matrix.target }}
      - name: Upload artifact
        uses: actions/upload-artifact@v4
        with:
          name: myapp-${{ matrix.target }}
          path: target/${{ matrix.target }}/release/myapp*

10. Database Migrations in Production

// Run migrations at startup before accepting traffic
use sqlx::PgPool;
use std::sync::Arc;
use tokio::sync::RwLock;

async fn run_startup(pool: &PgPool, ready: Arc<RwLock<bool>>) -> anyhow::Result<()> {
    tracing::info!("running database migrations…");

    // sqlx::migrate! embeds migrations from the migrations/ directory at compile time
    sqlx::migrate!("./migrations")
        .run(pool)
        .await
        .map_err(|e| anyhow::anyhow!("migration failed: {e}"))?;

    tracing::info!("migrations complete");

    // Signal that the application is ready to receive traffic
    *ready.write().await = true;
    Ok(())
}

// This avoids race conditions when multiple replicas start simultaneously
