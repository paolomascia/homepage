SQLx & Databases

Rust SQLx PostgreSQL

1. Setup

# Cargo.toml
[dependencies]
sqlx = { version = "0.8", features = [
    "runtime-tokio",   # async runtime
    "tls-rustls",      # TLS support
    "postgres",        # or "sqlite", "mysql"
    "uuid",            # UUID type support
    "chrono",          # DateTime type support
    "json",            # serde_json::Value
    "migrate",         # migrations
] }
tokio    = { version = "1", features = ["full"] }
serde    = { version = "1", features = ["derive"] }
uuid     = { version = "1", features = ["v4"] }
chrono   = { version = "0.4", features = ["serde"] }

# Install the sqlx CLI for running migrations
cargo install sqlx-cli --no-default-features --features rustls,postgres

# Create the database
export DATABASE_URL="postgres://user:pass@localhost/mydb"
sqlx database create

# Create a migration
sqlx migrate add create_users

# Run migrations
sqlx migrate run

# Revert the last migration
sqlx migrate revert

2. Connection Pool

use sqlx::postgres::PgPoolOptions;
use sqlx::PgPool;
use std::time::Duration;

async fn create_pool(database_url: &str) -> anyhow::Result<PgPool> {
    let pool = PgPoolOptions::new()
        .max_connections(20)               // max pool size
        .min_connections(2)                // keep 2 connections warm
        .acquire_timeout(Duration::from_secs(5))  // wait timeout for a connection
        .idle_timeout(Duration::from_secs(600))   // close idle connections after 10m
        .max_lifetime(Duration::from_secs(1800))  // recycle connections every 30m
        .connect(database_url)
        .await?;

    // Run pending migrations automatically on startup
    sqlx::migrate!("./migrations").run(&pool).await?;

    Ok(pool)
}

#[tokio::main]
async fn main() -> anyhow::Result<()> {
    let url = std::env::var("DATABASE_URL")
        .unwrap_or_else(|_| "postgres://postgres:password@localhost/mydb".to_string());

    let pool = create_pool(&url).await?;

    // Pool is Clone — pass it via Arc or directly (it's already Arc internally)
    println!("pool size: {}", pool.size());

    Ok(())
}

4. Compile-Time Checked Queries — query!

query! and query_as! verify your SQL against the live database at compile time. The type of every column is known statically — typos in column names are build errors.

use sqlx::PgPool;
use uuid::Uuid;
use chrono::{DateTime, Utc};
use serde::{Deserialize, Serialize};

#[derive(Debug, Serialize, Deserialize)]
struct User {
    id: Uuid,
    name: String,
    email: String,
    created_at: DateTime<Utc>,
    updated_at: DateTime<Utc>,
}

// query_as! maps rows directly to a struct — all column types are verified at compile time
async fn find_user(pool: &PgPool, id: Uuid) -> anyhow::Result<Option<User>> {
    let user = sqlx::query_as!(
        User,
        "SELECT id, name, email, created_at, updated_at
         FROM users
         WHERE id = $1",
        id
    )
    .fetch_optional(pool)
    .await?;

    Ok(user)
}

async fn find_user_by_email(pool: &PgPool, email: &str) -> anyhow::Result<Option<User>> {
    Ok(sqlx::query_as!(
        User,
        "SELECT id, name, email, created_at, updated_at
         FROM users WHERE email = $1",
        email
    )
    .fetch_optional(pool)
    .await?)
}

async fn list_users(pool: &PgPool, limit: i64, offset: i64) -> anyhow::Result<Vec<User>> {
    Ok(sqlx::query_as!(
        User,
        "SELECT id, name, email, created_at, updated_at
         FROM users
         ORDER BY created_at DESC
         LIMIT $1 OFFSET $2",
        limit,
        offset
    )
    .fetch_all(pool)
    .await?)
}

// For queries that return a single scalar value
async fn count_users(pool: &PgPool) -> anyhow::Result<i64> {
    let row = sqlx::query!("SELECT COUNT(*) as count FROM users")
        .fetch_one(pool)
        .await?;
    Ok(row.count.unwrap_or(0))
}

5. INSERT, UPDATE, DELETE

use sqlx::PgPool;
use uuid::Uuid;

#[derive(Debug)]
struct CreateUser {
    name: String,
    email: String,
}

async fn create_user(pool: &PgPool, input: CreateUser) -> anyhow::Result<User> {
    // RETURNING * gives us the full row back (including DB-generated fields)
    let user = sqlx::query_as!(
        User,
        "INSERT INTO users (name, email)
         VALUES ($1, $2)
         RETURNING id, name, email, created_at, updated_at",
        input.name,
        input.email,
    )
    .fetch_one(pool)
    .await?;

    Ok(user)
}

async fn update_user_name(pool: &PgPool, id: Uuid, name: &str) -> anyhow::Result<Option<User>> {
    Ok(sqlx::query_as!(
        User,
        "UPDATE users
         SET name = $1, updated_at = NOW()
         WHERE id = $2
         RETURNING id, name, email, created_at, updated_at",
        name,
        id,
    )
    .fetch_optional(pool)
    .await?)
}

async fn delete_user(pool: &PgPool, id: Uuid) -> anyhow::Result<bool> {
    let result = sqlx::query!(
        "DELETE FROM users WHERE id = $1",
        id
    )
    .execute(pool)
    .await?;

    Ok(result.rows_affected() == 1)
}

6. Transactions

use sqlx::{PgPool, postgres::PgConnection};
use uuid::Uuid;

// Transfer money between accounts atomically
async fn transfer(
    pool: &PgPool,
    from_id: Uuid,
    to_id: Uuid,
    amount: i64, // in cents
) -> anyhow::Result<()> {
    // begin() starts a transaction; commits on drop if no error
    let mut tx = pool.begin().await?;

    // Deduct from sender (lock the row to prevent races)
    let updated = sqlx::query!(
        "UPDATE accounts
         SET balance = balance - $1
         WHERE id = $2 AND balance >= $1
         RETURNING id",
        amount,
        from_id,
    )
    .fetch_optional(&mut *tx)
    .await?;

    if updated.is_none() {
        anyhow::bail!("insufficient funds or account not found");
    }

    // Add to recipient
    sqlx::query!(
        "UPDATE accounts SET balance = balance + $1 WHERE id = $2",
        amount,
        to_id,
    )
    .execute(&mut *tx)
    .await?;

    // Record the transfer
    sqlx::query!(
        "INSERT INTO transfers (from_id, to_id, amount) VALUES ($1, $2, $3)",
        from_id,
        to_id,
        amount,
    )
    .execute(&mut *tx)
    .await?;

    tx.commit().await?; // all-or-nothing commit
    Ok(())
}

// Savepoints for partial rollback
async fn bulk_insert_with_savepoint(pool: &PgPool, names: &[&str]) -> anyhow::Result<Vec<Uuid>> {
    let mut tx = pool.begin().await?;
    let mut ids = Vec::new();

    for &name in names {
        // Create a savepoint before each insert
        sqlx::query("SAVEPOINT insert_user").execute(&mut *tx).await?;

        match sqlx::query_scalar!(
            "INSERT INTO users (name, email) VALUES ($1, $2) RETURNING id",
            name,
            format!("{name}@example.com"),
        )
        .fetch_one(&mut *tx)
        .await {
            Ok(id) => ids.push(id),
            Err(e) => {
                // Roll back to savepoint — only this insert fails, not the whole tx
                sqlx::query("ROLLBACK TO SAVEPOINT insert_user")
                    .execute(&mut *tx).await?;
                eprintln!("skipped {name}: {e}");
            }
        }
    }

    tx.commit().await?;
    Ok(ids)
}

use sqlx::PgPool;
use futures::TryStreamExt;

async fn process_all_users(pool: &PgPool) -> anyhow::Result<()> {
    // fetch_many / fetch stream rows without loading everything into memory
    let mut stream = sqlx::query_as!(
        User,
        "SELECT id, name, email, created_at, updated_at FROM users ORDER BY created_at"
    )
    .fetch(pool); // returns a Stream

    let mut count = 0usize;
    while let Some(user) = stream.try_next().await? {
        // Process one user at a time — low memory footprint even for millions of rows
        process_user(&user).await;
        count += 1;
        if count % 1000 == 0 {
            println!("processed {count} users...");

8. Dynamic Queries with QueryBuilder

use sqlx::{PgPool, QueryBuilder};

#[derive(Default)]
struct UserFilter {
    name: Option<String>,
    email_domain: Option<String>,
    min_created: Option<chrono::DateTime<chrono::Utc>>,
    limit: Option<i64>,
}

async fn search_users(pool: &PgPool, filter: &UserFilter) -> anyhow::Result<Vec<User>> {
    let mut builder: QueryBuilder<sqlx::Postgres> = QueryBuilder::new(
        "SELECT id, name, email, created_at, updated_at FROM users WHERE 1=1"
    );

    if let Some(name) = &filter.name {
        builder.push(" AND name ILIKE ").push_bind(format!("%{name}%"));
    }
    if let Some(domain) = &filter.email_domain {
        builder.push(" AND email LIKE ").push_bind(format!("%@{domain}"));
    }
    if let Some(min) = filter.min_created {
        builder.push(" AND created_at >= ").push_bind(min);
    }

    builder.push(" ORDER BY created_at DESC");

    if let Some(limit) = filter.limit {
        builder.push(" LIMIT ").push_bind(limit);
    }

    Ok(builder.build_query_as::<User>()
        .fetch_all(pool)
        .await?)
}

9. Integration with Axum

use axum::{extract::{State, Path}, Json, Router, routing::{get, post}};
use sqlx::PgPool;
use uuid::Uuid;

// Pass the pool as Axum state
async fn api_get_user(
    State(pool): State<PgPool>,
    Path(id): Path<Uuid>,
) -> Result<Json<User>, (axum::http::StatusCode, String)> {
    find_user(&pool, id)
        .await
        .map_err(|e| (axum::http::StatusCode::INTERNAL_SERVER_ERROR, e.to_string()))?
        .map(Json)
        .ok_or((axum::http::StatusCode::NOT_FOUND, format!("user {id} not found")))
}

#[derive(serde::Deserialize)]
struct CreateUserBody { name: String, email: String }

async fn api_create_user(
    State(pool): State<PgPool>,
    Json(body): Json<CreateUserBody>,
) -> Result<(axum::http::StatusCode, Json<User>), (axum::http::StatusCode, String)> {
    let user = create_user(&pool, CreateUser { name: body.name, email: body.email })
        .await
        .map_err(|e| {
            if e.to_string().contains("unique") {
                (axum::http::StatusCode::CONFLICT, "email already in use".to_string())
            } else {
                (axum::http::StatusCode::INTERNAL_SERVER_ERROR, e.to_string())
            }
        })?;

    Ok((axum::http::StatusCode::CREATED, Json(user)))
}

#[tokio::main]
async fn main() -> anyhow::Result<()> {
    let url = std::env::var("DATABASE_URL")?;
    let pool = create_pool(&url).await?;

    let app = Router::new()
        .route("/users/:id", get(api_get_user))
        .route("/users",     post(api_create_user))
        .with_state(pool);

    let listener = tokio::net::TcpListener::bind("0.0.0.0:3000").await?;
    axum::serve(listener, app).await?;
    Ok(())
}

10. Testing with a Real Database

# Cargo.toml
[dev-dependencies]
sqlx = { version = "0.8", features = ["test-utils"] }

#[cfg(test)]
mod tests {
    use super::*;
    use sqlx::PgPool;

    // sqlx::test spins up a fresh DB (via Docker or an existing PG instance),
    // runs migrations, and gives you an isolated pool per test.
    // Set TEST_DATABASE_URL or use sqlx::test(migrations = "migrations")
    #[sqlx::test(migrations = "./migrations")]
    async fn test_create_and_find_user(pool: PgPool) {
        let user = create_user(&pool, CreateUser {
            name: "Alice".to_string(),
            email: "alice@test.com".to_string(),
        })
        .await
        .unwrap();

        assert_eq!(user.name, "Alice");
        assert_eq!(user.email, "alice@test.com");

        let found = find_user(&pool, user.id).await.unwrap();
        assert!(found.is_some());
        assert_eq!(found.unwrap().name, "Alice");
    }

    #[sqlx::test(migrations = "./migrations")]
    async fn test_delete_user(pool: PgPool) {
        let user = create_user(&pool, CreateUser {
            name: "Bob".to_string(),
            email: "bob@test.com".to_string(),
        })
        .await
        .unwrap();

        let deleted = delete_user(&pool, user.id).await.unwrap();
        assert!(deleted);

        let not_found = find_user(&pool, user.id).await.unwrap();
        assert!(not_found.is_none());
    }
}
