2. WebSocket Handler with Axum

use axum::{
    extract::ws::{WebSocket, WebSocketUpgrade, Message},
    response::IntoResponse,
    routing::get,
    Router,
};
use futures::{SinkExt, StreamExt};

// The upgrade handler — HTTP → WebSocket handshake
async fn ws_handler(ws: WebSocketUpgrade) -> impl IntoResponse {
    ws.on_upgrade(handle_socket)
}

// Once upgraded, `socket` is a bidirectional async stream of Messages
async fn handle_socket(mut socket: WebSocket) {
    // Send a welcome message
    if socket.send(Message::Text("Welcome!".into())).await.is_err() {
        return; // client disconnected
    }

    // Echo loop
    while let Some(msg) = socket.recv().await {
        match msg {
            Ok(Message::Text(text)) => {
                println!("received: {text}");
                let reply = format!("echo: {text}");
                if socket.send(Message::Text(reply.into())).await.is_err() {
                    break; // client disconnected
                }
            }
            Ok(Message::Binary(data)) => {
                println!("binary: {} bytes", data.len());
                let _ = socket.send(Message::Binary(data)).await;
            }
            Ok(Message::Ping(data)) => {
                let _ = socket.send(Message::Pong(data)).await;
            }
            Ok(Message::Close(_)) | Err(_) => break,
            Ok(Message::Pong(_)) => {} // ignore
        }
    }

    println!("client disconnected");
}

#[tokio::main]
async fn main() {
    let app = Router::new()
        .route("/ws", get(ws_handler));

    let listener = tokio::net::TcpListener::bind("0.0.0.0:3000").await.unwrap();
    println!("WebSocket server on ws://localhost:3000/ws");
    axum::serve(listener, app).await.unwrap();
}

3. Split Send / Receive

Split the socket into separate sender and receiver halves to drive them independently — useful for concurrent reading and writing.

use axum::extract::ws::{WebSocket, Message};
use futures::{SinkExt, StreamExt};
use tokio::time::{interval, Duration};

async fn handle_socket_split(socket: WebSocket) {
    let (mut sender, mut receiver) = socket.split();

    // Spawn a task that sends a heartbeat every 30 seconds
    let mut ping_task = tokio::spawn(async move {
        let mut ticker = interval(Duration::from_secs(30));
        loop {
            ticker.tick().await;
            if sender.send(Message::Ping(vec![].into())).await.is_err() {
                break;
            }
        }
    });

    // Read messages in the current task
    let mut recv_task = tokio::spawn(async move {
        while let Some(msg) = receiver.next().await {
            match msg {
                Ok(Message::Text(t)) => println!("client: {t}"),
                Ok(Message::Close(_)) | Err(_) => break,
                _ => {}
            }
        }
    });

    // If either task finishes, abort the other
    tokio::select! {
        _ = &mut ping_task => recv_task.abort(),
        _ = &mut recv_task => ping_task.abort(),
    }
}

4. Broadcast — WebSocket Chat Server

Use tokio::sync::broadcast to fan out messages from any client to all connected clients.

use axum::{
    extract::{ws::{WebSocket, WebSocketUpgrade, Message}, State},
    response::IntoResponse,
    routing::get,
    Router,
};
use futures::{SinkExt, StreamExt};
use serde::{Deserialize, Serialize};
use std::sync::Arc;
use tokio::sync::broadcast;

#[derive(Clone, Debug, Serialize, Deserialize)]
struct ChatMessage {
    sender: String,
    content: String,
    timestamp: u64,
}

#[derive(Clone)]
struct AppState {
    tx: broadcast::Sender<ChatMessage>,
}

async fn chat_handler(
    ws: WebSocketUpgrade,
    State(state): State<AppState>,
) -> impl IntoResponse {
    ws.on_upgrade(|socket| handle_chat(socket, state))
}

async fn handle_chat(socket: WebSocket, state: AppState) {
    let (mut sender, mut receiver) = socket.split();

    // Each connection gets its own receiver from the broadcast channel
    let mut rx = state.tx.subscribe();

    // Task: forward broadcast messages to this WebSocket client
    let mut send_task = tokio::spawn(async move {
        while let Ok(msg) = rx.recv().await {
            let json = serde_json::to_string(&msg).unwrap();
            if sender.send(Message::Text(json.into())).await.is_err() {
                break;
            }
        }
    });

    // Task: read from this WebSocket client and broadcast to all others
    let tx = state.tx.clone();
    let mut recv_task = tokio::spawn(async move {
        while let Some(Ok(Message::Text(text))) = receiver.next().await {
            // Parse the incoming message
            if let Ok(mut msg) = serde_json::from_str::<ChatMessage>(&text) {
                msg.timestamp = std::time::SystemTime::now()
                    .duration_since(std::time::UNIX_EPOCH)
                    .unwrap()
                    .as_secs();

                // Broadcast to ALL subscribers (including the sender)
                let _ = tx.send(msg);
            }
        }
    });

    tokio::select! {
        _ = &mut send_task => recv_task.abort(),
        _ = &mut recv_task => send_task.abort(),
    }
}

#[tokio::main]
async fn main() {
    let (tx, _rx) = broadcast::channel::<ChatMessage>(128);
    let state = AppState { tx };

    let app = Router::new()
        .route("/ws/chat", get(chat_handler))
        .with_state(state);

    let listener = tokio::net::TcpListener::bind("0.0.0.0:3000").await.unwrap();
    println!("Chat server on ws://localhost:3000/ws/chat");
    axum::serve(listener, app).await.unwrap();
}

5. Rooms — Targeted Broadcast

use std::collections::HashMap;
use std::sync::{Arc, RwLock};
use tokio::sync::broadcast;

type RoomMap = Arc<RwLock<HashMap<String, broadcast::Sender<String>>>>;

fn get_or_create_room(rooms: &RoomMap, room_id: &str) -> broadcast::Sender<String> {
    // Fast path: room already exists
    {
        let map = rooms.read().unwrap();
        if let Some(tx) = map.get(room_id) {
            return tx.clone();
        }
    }

    // Slow path: create the room
    let mut map = rooms.write().unwrap();
    // Double-check in case another thread created it while we waited
    map.entry(room_id.to_string())
        .or_insert_with(|| broadcast::channel(64).0)
        .clone()
}

async fn handle_room_socket(
    socket: WebSocket,
    rooms: RoomMap,
    room_id: String,
    user_id: String,
) {
    let tx = get_or_create_room(&rooms, &room_id);
    let mut rx = tx.subscribe();
    let (mut sender, mut receiver) = socket.split();

    // Announce join
    let _ = tx.send(format!("{user_id} joined room {room_id}"));

    let user_id_clone = user_id.clone();
    let tx_clone = tx.clone();

    let mut send_task = tokio::spawn(async move {
        while let Ok(msg) = rx.recv().await {
            if sender.send(Message::Text(msg.into())).await.is_err() {
                break;
            }
        }
    });

    let mut recv_task = tokio::spawn(async move {
        while let Some(Ok(Message::Text(text))) = receiver.next().await {
            let broadcast_msg = format!("[{user_id_clone}] {text}");
            let _ = tx_clone.send(broadcast_msg);
        }
    });

    tokio::select! {
        _ = &mut send_task => recv_task.abort(),
        _ = &mut recv_task => send_task.abort(),
    }

    // Announce leave
    let _ = tx.send(format!("{user_id} left room {room_id}"));
}

6. Connection Metadata and Query Params

use axum::{
    extract::{ws::WebSocketUpgrade, Query, State},
    response::IntoResponse,
};
use serde::Deserialize;

#[derive(Deserialize)]
struct ConnectParams {
    token: String,
    room: Option<String>,
}

async fn authenticated_ws(
    ws: WebSocketUpgrade,
    Query(params): Query<ConnectParams>,
    State(state): State<AppState>,
) -> impl IntoResponse {
    // Validate the token before upgrading
    if params.token != "valid-token" {
        return axum::http::StatusCode::UNAUTHORIZED.into_response();
    }

    let room = params.room.unwrap_or_else(|| "general".to_string());

    ws.on_upgrade(move |socket| async move {
        handle_chat(socket, state).await;
    })
}

7. tokio-tungstenite WebSocket Client

use tokio_tungstenite::{connect_async, tungstenite::Message};
use futures::{SinkExt, StreamExt};
use serde_json::json;

#[tokio::main]
async fn main() -> anyhow::Result<()> {
    let url = "ws://localhost:3000/ws/chat";

    let (ws_stream, _) = connect_async(url).await?;
    println!("connected to {url}");

    let (mut write, mut read) = ws_stream.split();

    // Send a message
    let msg = json!({
        "sender": "bot",
        "content": "Hello from Rust client!",
        "timestamp": 0
    });
    write.send(Message::Text(msg.to_string())).await?;

    // Receive messages
    let recv_task = tokio::spawn(async move {
        while let Some(msg) = read.next().await {
            match msg {
                Ok(Message::Text(text)) => println!("<< {text}"),
                Ok(Message::Close(_)) => break,
                Err(e) => { eprintln!("error: {e}"); break; }
                _ => {}
            }
        }
    });

    // Send a few more messages then close
    for i in 1..=3 {
        tokio::time::sleep(tokio::time::Duration::from_millis(500)).await;
        write.send(Message::Text(format!("message {i}"))).await?;
    }
    write.send(Message::Close(None)).await?;

    recv_task.await?;
    Ok(())
}

8. Complete: Chat Server + HTML Client

// Server: serves the HTML page AND handles WebSocket connections
use axum::{routing::get, Router};
use tower_http::services::ServeDir;

async fn full_server() {
    let (tx, _) = tokio::sync::broadcast::channel::<ChatMessage>(128);
    let state = AppState { tx };

    let app = Router::new()
        .route("/ws/chat", get(chat_handler))
        // Serve static HTML/JS from the public/ directory
        .nest_service("/", ServeDir::new("public"))
        .with_state(state);

    let listener = tokio::net::TcpListener::bind("0.0.0.0:3000").await.unwrap();
    println!("open http://localhost:3000");
    axum::serve(listener, app).await.unwrap();
}

9. Production Considerations

  • Heartbeat / ping-pong: send a ping every 30s; close connections that don't respond within the timeout.
  • Message size limits: reject oversized frames to prevent memory exhaustion.
  • Back-pressure: use bounded channels (mpsc with capacity) per connection so a slow client doesn't block others.
  • Authentication: validate the token in the HTTP upgrade request (query param or Sec-WebSocket-Protocol header) before upgrading.
  • Horizontal scaling: use Redis pub/sub (via redis crate) as the broadcast backbone so multiple server instances share messages.
  • Reconnection on the client: implement exponential-backoff reconnect in the JavaScript client.
// Per-connection message size limit
async fn handle_socket_safe(ws: WebSocketUpgrade) -> impl IntoResponse {
    ws.max_message_size(64 * 1024)   // reject frames > 64 KiB
      .max_frame_size(16 * 1024)     // reject individual frames > 16 KiB
      .on_upgrade(handle_socket)
}
