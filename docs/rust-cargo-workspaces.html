Cargo Workspaces, Features & Build Scripts

Rust Cargo Build System

1. Cargo Workspaces

A workspace is a directory containing multiple related crates that share a single Cargo.lock and build output directory. This avoids redundant compilation and keeps dependency versions consistent.

1.1 Workspace Layout

my-project/                    ← workspace root
├── Cargo.toml                 ← workspace manifest
├── Cargo.lock                 ← shared lockfile (one per workspace)
├── crates/
│   ├── core/                  ← library crate
│   │   ├── Cargo.toml
│   │   └── src/lib.rs
│   ├── cli/                   ← binary crate
│   │   ├── Cargo.toml
│   │   └── src/main.rs
│   └── derive/                ← proc-macro crate
│       ├── Cargo.toml
│       └── src/lib.rs
└── target/                    ← shared build output

# Cargo.toml (workspace root)
[workspace]
members = [
    "crates/core",
    "crates/cli",
    "crates/derive",
]
resolver = "2"   # use the v2 feature resolver (required for Rust 2021+)

# Shared dependency versions — avoids repetition in member Cargo.toml files
[workspace.dependencies]
serde = { version = "1", features = ["derive"] }
tokio = { version = "1", features = ["full"] }
anyhow = "1"

# crates/core/Cargo.toml
[package]
name = "my-core"
version = "0.1.0"
edition = "2021"

[dependencies]
# Reference workspace-level dependency — version is inherited
serde = { workspace = true }
anyhow = { workspace = true }

# crates/cli/Cargo.toml
[package]
name = "my-cli"
version = "0.1.0"
edition = "2021"

[dependencies]
# Reference another workspace member
my-core = { path = "../core" }
tokio = { workspace = true }
clap = { version = "4", features = ["derive"] }

1.2 Working with Workspaces

# Build all crates in the workspace
cargo build --workspace

# Build a specific member
cargo build -p my-cli

# Run tests for all members
cargo test --workspace

# Run a specific binary
cargo run -p my-cli -- --help

# Check all crates without building
cargo check --workspace

# Update dependencies for all members
cargo update

1.3 Workspace Metadata

# Cargo.toml (workspace root)
[workspace]
members = ["crates/*"]  # glob pattern — includes all subdirectories of crates/
exclude = ["crates/experiments"]

# Shared metadata — members can inherit with `package.authors.workspace = true`
[workspace.package]
authors = ["Paolo Mascia"]
license = "MIT OR Apache-2.0"
repository = "https://github.com/user/my-project"
edition = "2021"

2. Feature Flags

Features let users opt into optional functionality. They control which dependencies are compiled and which code is included.

2.1 Defining Features

# Cargo.toml
[package]
name = "my-crate"
version = "0.1.0"

[features]
# Default features (enabled unless --no-default-features is used)
default = ["std"]

# Feature groups
std    = []
async  = ["dep:tokio", "dep:futures"]
serde  = ["dep:serde"]
full   = ["async", "serde"]

# Mutually exclusive features (enforce at runtime or doc them clearly)
backend-sqlite  = ["dep:rusqlite"]
backend-postgres = ["dep:postgres"]

[dependencies]
# Optional dependency — only compiled when the "async" feature is active
tokio   = { version = "1", features = ["full"], optional = true }
futures = { version = "0.3", optional = true }
serde   = { version = "1", features = ["derive"], optional = true }
rusqlite = { version = "0.31", optional = true }
postgres = { version = "0.19", optional = true }

2.2 Using Features in Code

// Feature-conditional code with cfg attributes
#[cfg(feature = "serde")]
use serde::{Serialize, Deserialize};

#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
#[derive(Debug)]
pub struct Config {
    pub host: String,
    pub port: u16,
}

impl Config {
    pub fn new(host: impl Into<String>, port: u16) -> Self {
        Config { host: host.into(), port }
    }

    // Only available with the "async" feature
    #[cfg(feature = "async")]
    pub async fn connect(&self) -> Result<tokio::net::TcpStream, std::io::Error> {
        tokio::net::TcpStream::connect(format!("{}:{}", self.host, self.port)).await
    }

    // Only available with the "std" feature
    #[cfg(feature = "std")]
    pub fn from_env() -> Option<Self> {
        Some(Config {
            host: std::env::var("APP_HOST").ok()?,
            port: std::env::var("APP_PORT").ok()?.parse().ok()?,
        })
    }
}

// Use cfg!() as a runtime bool
fn log_level() -> &'static str {
    if cfg!(debug_assertions) { "debug" } else { "info" }
}

2.3 Testing Features

# Test with no default features
cargo test --no-default-features

# Test with specific features
cargo test --features "async,serde"

# Test all feature combinations (use cargo-hack)
cargo install cargo-hack
cargo hack test --each-feature        # test each feature in isolation
cargo hack test --feature-powerset   # test every combination (slow!)

# Check which features are enabled for a dependency
cargo tree --edges features -p my-crate

3. Conditional Compilation

// Platform-specific code
#[cfg(target_os = "windows")]
fn get_home() -> std::path::PathBuf {
    std::env::var("USERPROFILE").unwrap().into()
}

#[cfg(target_os = "linux")]
fn get_home() -> std::path::PathBuf {
    std::env::var("HOME").unwrap().into()
}

#[cfg(target_os = "macos")]
fn get_home() -> std::path::PathBuf {
    std::env::var("HOME").unwrap().into()
}

// Architecture-specific
#[cfg(target_arch = "x86_64")]
fn fast_hash(data: &[u8]) -> u64 {
    // use SSE/AVX intrinsics
    0 // placeholder
}

#[cfg(not(target_arch = "x86_64"))]
fn fast_hash(data: &[u8]) -> u64 {
    // portable fallback
    data.iter().fold(0u64, |acc, &b| acc.wrapping_mul(31).wrapping_add(b as u64))
}

// Multiple conditions with any/all/not
#[cfg(all(target_os = "linux", target_arch = "x86_64"))]
fn linux_x86_only() {}

#[cfg(any(target_os = "linux", target_os = "macos"))]
fn unix_only() {}

#[cfg(not(windows))]
fn not_windows() {}

// Debug vs release builds
#[cfg(debug_assertions)]
fn validate_invariant(x: i32) {
    assert!(x >= 0, "invariant violated: x must be non-negative");
}
#[cfg(not(debug_assertions))]
fn validate_invariant(_: i32) {} // no-op in release

4. Build Scripts (build.rs)

A build.rs file at the crate root runs before compilation. It's a regular Rust program that can generate code, set environment variables, link native libraries, and more.

4.1 Code Generation

// build.rs — generates Rust code from a data file at build time
use std::env;
use std::fs;
use std::path::Path;

fn main() {
    let out_dir = env::var("OUT_DIR").unwrap();
    let dest = Path::new(&out_dir).join("generated.rs");

    // Generate a lookup table from a text file
    let data = fs::read_to_string("src/keywords.txt").unwrap_or_default();
    let entries: Vec<&str> = data.lines().filter(|l| !l.is_empty()).collect();

    let mut code = String::from("pub const KEYWORDS: &[&str] = &[\n");
    for kw in &entries {
        code.push_str(&format!("    \"{kw}\",\n"));
    }
    code.push_str("];\n");

    fs::write(&dest, code).unwrap();

    // Tell Cargo to re-run build.rs only if this file changes
    println!("cargo:rerun-if-changed=src/keywords.txt");
    println!("cargo:rerun-if-changed=build.rs");
}

// src/lib.rs — include the generated file
include!(concat!(env!("OUT_DIR"), "/generated.rs"));

pub fn is_keyword(s: &str) -> bool {
    KEYWORDS.contains(&s)
}

4.2 Linking Native Libraries

// build.rs — linking a native C library
fn main() {
    // Link against libssl and libcrypto
    println!("cargo:rustc-link-lib=ssl");
    println!("cargo:rustc-link-lib=crypto");

    // Search path for the library
    println!("cargo:rustc-link-search=/usr/local/lib");

    // Or use pkg-config (add pkg-config crate as build dependency)
    // pkg_config::probe_library("openssl").unwrap();

    // Pass a flag to the compiler
    println!("cargo:rustc-cfg=has_openssl");

    // Set an environment variable accessible in source via env!()
    println!("cargo:rustc-env=BUILD_TIMESTAMP=2026-03-01");

    println!("cargo:rerun-if-changed=build.rs");
}

// src/lib.rs — using the build-time environment variable
pub const BUILD_TIMESTAMP: &str = env!("BUILD_TIMESTAMP");

pub fn version_info() -> String {
    format!("v{} built on {}", env!("CARGO_PKG_VERSION"), BUILD_TIMESTAMP)
}

4.3 Compiling C Code with cc

# Cargo.toml
[build-dependencies]
cc = "1"

// build.rs
fn main() {
    // Compile C source files into a static library
    cc::Build::new()
        .file("src/native/fast_hash.c")
        .file("src/native/simd_ops.c")
        .flag_if_supported("-O3")
        .flag_if_supported("-march=native")
        .compile("fast_native"); // outputs libfast_native.a

    println!("cargo:rerun-if-changed=src/native/fast_hash.c");
    println!("cargo:rerun-if-changed=src/native/simd_ops.c");
}

// src/lib.rs — calling the compiled C function
extern "C" {
    fn fast_hash_c(data: *const u8, len: usize) -> u64;
}

pub fn fast_hash(data: &[u8]) -> u64 {
    unsafe { fast_hash_c(data.as_ptr(), data.len()) }
}

4.4 Useful build.rs Variables

fn main() {
    // Cargo-provided environment variables available in build.rs
    let pkg_name    = env!("CARGO_PKG_NAME");
    let pkg_version = env!("CARGO_PKG_VERSION");
    let manifest    = env!("CARGO_MANIFEST_DIR"); // crate root directory
    let out_dir     = std::env::var("OUT_DIR").unwrap(); // temp output dir
    let profile     = std::env::var("PROFILE").unwrap(); // "debug" or "release"
    let target      = std::env::var("TARGET").unwrap();  // "x86_64-unknown-linux-gnu"
    let host        = std::env::var("HOST").unwrap();    // build machine target

    println!("building {} v{} for {}", pkg_name, pkg_version, target);

    // Available in source code via env!() macro
    println!("cargo:rustc-env=TARGET_TRIPLE={target}");
    println!("cargo:rustc-env=BUILD_PROFILE={profile}");
}

5. Profile Configuration

# Cargo.toml
[profile.dev]
opt-level = 0        # no optimizations (fast compile, slow run)
debug = true         # include debug symbols
overflow-checks = true

[profile.release]
opt-level = 3        # full optimizations
debug = false
lto = true           # link-time optimization (slower link, faster binary)
codegen-units = 1    # one codegen unit (better optimization, slower compile)
panic = "abort"      # abort on panic instead of unwind (smaller binary)
strip = "symbols"    # strip debug symbols from binary

# Custom profile for profiling (release + debug symbols)
[profile.profiling]
inherits = "release"
debug = true
strip = false

# Dev builds of dependencies can be optimized even in dev mode
[profile.dev.package."*"]
opt-level = 2

# Build with a custom profile
cargo build --profile profiling

# Check binary size
cargo build --release
ls -lh target/release/my-binary

# Analyze what's in the binary
cargo install cargo-bloat
cargo bloat --release -n 20  # top 20 largest functions

6. Publishing to crates.io

# Cargo.toml — required fields for publishing
[package]
name = "my-awesome-crate"
version = "1.0.0"
edition = "2021"
authors = ["Paolo Mascia <paolo@example.com>"]
description = "A short, catchy description"
license = "MIT OR Apache-2.0"
repository = "https://github.com/user/my-awesome-crate"
homepage = "https://my-awesome-crate.rs"
documentation = "https://docs.rs/my-awesome-crate"
readme = "README.md"
keywords = ["networking", "async", "tokio"] # max 5
categories = ["network-programming"]        # from crates.io category list
exclude = ["tests/fixtures/**", "benches/**"] # don't include in the package

# Login to crates.io (get token from https://crates.io/me)
cargo login <your-api-token>

# Verify the package before publishing (dry run)
cargo publish --dry-run

# See what will be included in the package
cargo package --list

# Publish (cannot be undone — only yanking is possible)
cargo publish

# Yank a version (it still exists but cannot be added as a new dep)
cargo yank --version 1.0.0
cargo yank --version 1.0.0 --undo   # un-yank

# Update the version and publish a patch
cargo set-version 1.0.1             # or edit Cargo.toml manually
cargo publish

6.1 Semantic Versioning

MAJOR.MINOR.PATCH

PATCH: bug fix, no API change               → 1.0.0 → 1.0.1
MINOR: new public API, backwards compatible → 1.0.1 → 1.1.0
MAJOR: breaking API change                  → 1.1.0 → 2.0.0

Pre-1.0: minor = breaking, patch = backwards compat
         0.1.0 → 0.2.0 (breaking), 0.1.0 → 0.1.1 (patch)

Dependency version specifiers in Cargo.toml:
  "1"        → >=1.0.0, <2.0.0   (caret, default)
  "^1.2"     → >=1.2.0, <2.0.0
  "~1.2.3"   → >=1.2.3, <1.3.0   (tilde, only patch updates)
  "=1.2.3"   → exactly 1.2.3
  "*"        → any version        (avoid!)
  ">=1, <3"  → range

7. Essential Cargo Commands

# ── Project management ──────────────────────────────────────────────────
cargo new my-project          # new binary crate
cargo new --lib my-library    # new library crate
cargo init                    # init in current directory

# ── Building ─────────────────────────────────────────────────────────────
cargo build                   # debug build
cargo build --release         # release build
cargo check                   # type-check only (fast)
cargo clippy                  # run linter
cargo fix                     # auto-apply clippy suggestions
cargo fmt                     # format code

# ── Dependencies ─────────────────────────────────────────────────────────
cargo add serde --features derive    # add dependency
cargo add tokio@1 --features full    # specific version
cargo remove serde                   # remove dependency
cargo update                         # update Cargo.lock
cargo tree                           # show dependency tree
cargo tree --duplicates              # find duplicate versions

# ── Documentation ────────────────────────────────────────────────────────
cargo doc --open              # build and open docs
cargo doc --no-deps           # only document current crate

# ── Useful extensions ────────────────────────────────────────────────────
cargo install cargo-expand    # expand macros
cargo install cargo-watch     # re-run on file change
cargo install cargo-audit     # security vulnerability scan
cargo install cargo-outdated  # find outdated dependencies
cargo install cargo-bloat     # analyze binary size
cargo install cargo-hack      # test feature combinations
cargo install cargo-llvm-cov  # code coverage

# cargo-watch examples
cargo watch -x check          # check on every save
cargo watch -x test           # test on every save
cargo watch -x 'run -- --port 8080'

8. Advanced Workspace Tips

8.1 Virtual Workspaces

# A "virtual workspace" has no [package] section — the root is only for coordination
# This is common for monorepos where all crates are in subdirectories

# Cargo.toml (virtual workspace root)
[workspace]
members = ["crates/*", "tools/*"]
resolver = "2"

# No [package] — the root itself is not a crate

8.2 Path Dependencies in Development

# Use path = during development, swap for version = when publishing
[dependencies]
# Development: use local path
my-lib = { path = "../my-lib" }

# Or with a version for publishing (both path and version = use path locally)
my-lib = { path = "../my-lib", version = "0.1" }

8.3 Workspace Script Runner

# Run the same command across all workspace members
cargo test --workspace --all-features
cargo clippy --workspace --all-targets -- -D warnings
cargo fmt --all

# Build only the changed crates (use cargo-workspaces)
cargo install cargo-workspaces
cargo workspaces list         # list all members
cargo workspaces publish      # version-bump and publish all

8.4 Per-Crate .cargo/config.toml

# .cargo/config.toml (workspace root) — applies to all members
[build]
rustflags = ["-C", "target-cpu=native"]  # optimize for current CPU

[alias]
t    = "test"
b    = "build --release"
c    = "clippy --all-targets -- -D warnings"
lint = "clippy --workspace --all-targets --all-features -- -D warnings"

[target.x86_64-unknown-linux-musl]
linker = "x86_64-linux-musl-gcc"         # static linking
rustflags = ["-C", "target-feature=+crt-static"]

# Use a faster linker in development
[target.x86_64-unknown-linux-gnu]
linker = "clang"
rustflags = ["-C", "link-arg=-fuse-ld=lld"]
