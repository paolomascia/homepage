OS Threads & Scoped Threads in Rust

Rust Threads Concurrency

1. thread::spawn and JoinHandle

std::thread::spawn launches an OS thread. It takes a closure and returns a JoinHandle<T> that you can .join() to wait for the thread and retrieve its return value.

use std::thread;
use std::time::Duration;

fn main() {
    // spawn returns JoinHandle<()>
    let handle = thread::spawn(|| {
        for i in 0..5 {
            println!("spawned thread: {i}");
            thread::sleep(Duration::from_millis(10));
        }
        42 // return value
    });

    for i in 0..3 {
        println!("main thread: {i}");
        thread::sleep(Duration::from_millis(15));
    }

    // join() blocks until the thread finishes; returns Result<T, Box<dyn Any>>
    let result = handle.join().expect("thread panicked");
    println!("thread returned: {result}"); // 42
}

If the spawned thread panics, join() returns Err. You can check and recover:

use std::thread;

fn main() {
    let h = thread::spawn(|| panic!("oops"));

    match h.join() {
        Ok(val) => println!("ok: {val:?}"),
        Err(e)  => println!("thread panicked: {:?}", e.downcast_ref::<&str>()),
    }
}

2. move Closures — Transferring Ownership to Threads

Spawned threads must have 'static lifetimes for captured data because the thread may outlive the current scope. Use move to transfer ownership into the closure.

use std::thread;

fn main() {
    let data = vec![1, 2, 3, 4, 5];

    // Without `move` this would not compile: `data` might not live long enough
    let handle = thread::spawn(move || {
        // `data` is moved into the thread
        let sum: i32 = data.iter().sum();
        println!("sum = {sum}");
        sum
    });

    // `data` is no longer accessible here — it was moved
    let result = handle.join().unwrap();
    println!("result: {result}");
}

To share data with multiple threads, use Arc (covered in smart-pointers) to clone a reference-counted pointer before moving it.

use std::sync::Arc;
use std::thread;

fn main() {
    let data = Arc::new(vec![10, 20, 30]);
    let mut handles = Vec::new();

    for i in 0..3 {
        let data = Arc::clone(&data); // cheap clone of the Arc
        handles.push(thread::spawn(move || {
            println!("thread {i}: data[{i}] = {}", data[i]);
        }));
    }

    for h in handles {
        h.join().unwrap();
    }
}

3. Scoped Threads — Borrowing without Arc

thread::scope (stable since Rust 1.63) creates a scope that guarantees all spawned threads finish before the scope exits, allowing threads to borrow local data without Arc.

use std::thread;

fn main() {
    let data = vec![1, 2, 3, 4, 5, 6, 7, 8];
    let chunk_size = data.len() / 2;

    // thread::scope guarantees all spawned threads join before it returns
    thread::scope(|s| {
        for chunk in data.chunks(chunk_size) {
            s.spawn(|| {
                // `chunk` is a borrowed slice — no Arc needed!
                let sum: i32 = chunk.iter().sum();
                println!("chunk sum = {sum}");
            });
        }
        // all threads join here automatically
    });

    // `data` is still accessible after the scope
    println!("data is still here: {data:?}");
}

Scoped threads are the preferred way to parallelize work on local data. They eliminate the need for Arc and are more ergonomic than manual join.

3.1 Mutable Borrows in Scopes

Each thread needs a disjoint mutable borrow — use split_at_mut or process chunks separately:

use std::thread;

fn parallel_increment(data: &mut [i32]) {
    // split_at_mut gives two non-overlapping mutable borrows
    let mid = data.len() / 2;
    let (left, right) = data.split_at_mut(mid);

    thread::scope(|s| {
        s.spawn(|| {
            for x in left.iter_mut() { *x += 1; }
        });
        s.spawn(|| {
            for x in right.iter_mut() { *x += 10; }
        });
    });
}

fn main() {
    let mut v = vec![0, 1, 2, 3, 4, 5];
    parallel_increment(&mut v);
    println!("{v:?}"); // [1, 2, 3, 13, 14, 15]
}

4. Thread Builder — Names and Stack Size

Use thread::Builder to name threads (visible in panic messages and debuggers) and to set a custom stack size.

use std::thread;

fn main() {
    let handle = thread::Builder::new()
        .name("worker-1".to_string())
        .stack_size(4 * 1024 * 1024) // 4 MiB
        .spawn(|| {
            // thread::current().name() is Some("worker-1")
            let name = thread::current().name().unwrap_or("unnamed").to_string();
            println!("I am thread: {name}");
            // simulate work
        })
        .expect("failed to spawn thread");

    handle.join().unwrap();
}

5. Thread Parking

thread::park() suspends the current thread until another thread calls unpark() on its handle. This is a low-level synchronization primitive — prefer channels or Condvar for most use cases.

use std::thread;
use std::time::Duration;

fn main() {
    let worker = thread::Builder::new()
        .name("worker".to_string())
        .spawn(|| {
            println!("worker: starting, parking...");
            thread::park(); // suspend until unpark()
            println!("worker: unparked, doing work");
            thread::park(); // park again
            println!("worker: unparked again, finishing");
        })
        .unwrap();

    thread::sleep(Duration::from_millis(100));
    println!("main: unparking worker");
    worker.thread().unpark();

    thread::sleep(Duration::from_millis(100));
    println!("main: unparking worker again");
    worker.thread().unpark();

    worker.join().unwrap();
}
// Output (order may vary):
// worker: starting, parking...
// main: unparking worker
// worker: unparked, doing work
// main: unparking worker again
// worker: unparked again, finishing

Note: park() can spuriously wake up, so always check a condition in a loop:

use std::sync::atomic::{AtomicBool, Ordering};
use std::sync::Arc;
use std::thread;

fn main() {
    let ready = Arc::new(AtomicBool::new(false));
    let ready2 = Arc::clone(&ready);
    let main_thread = thread::current();

    let worker = thread::spawn(move || {
        // loop guards against spurious wakeups
        while !ready2.load(Ordering::Acquire) {
            thread::park();
        }
        println!("worker: condition satisfied, proceeding");
    });

    thread::sleep(std::time::Duration::from_millis(50));
    ready.store(true, Ordering::Release);
    worker.thread().unpark();
    worker.join().unwrap();
}

6. Thread-Local Storage

The thread_local! macro creates storage that is private to each thread — each thread gets its own instance, initialized on first access.

use std::cell::RefCell;
use std::thread;

// Each thread gets its own Vec<String>
thread_local! {
    static LOG: RefCell<Vec<String>> = RefCell::new(Vec::new());
}

fn log(msg: &str) {
    LOG.with(|l| l.borrow_mut().push(msg.to_string()));
}

fn flush() -> Vec<String> {
    LOG.with(|l| l.borrow().clone())
}

fn main() {
    let h1 = thread::spawn(|| {
        log("h1: step 1");
        log("h1: step 2");
        flush()
    });

    let h2 = thread::spawn(|| {
        log("h2: only step");
        flush()
    });

    log("main: step 1");

    println!("h1 log: {:?}", h1.join().unwrap());
    println!("h2 log: {:?}", h2.join().unwrap());
    println!("main log: {:?}", flush());
    // Each thread sees only its own entries
}

7. Shared Mutable State — Arc<Mutex<T>>

When multiple threads need to mutate shared state, wrap the data in Arc<Mutex<T>>. Keep lock guards short-lived to maximize concurrency.

use std::sync::{Arc, Mutex};
use std::thread;

fn main() {
    let counter = Arc::new(Mutex::new(0u64));
    let mut handles = Vec::new();

    for _ in 0..8 {
        let counter = Arc::clone(&counter);
        handles.push(thread::spawn(move || {
            for _ in 0..1_000 {
                // lock() blocks until the mutex is available
                // the guard is dropped at the end of the block
                let mut n = counter.lock().unwrap();
                *n += 1;
                // guard dropped here — mutex released immediately
            }
        }));
    }

    for h in handles { h.join().unwrap(); }

    println!("counter = {}", counter.lock().unwrap()); // 8000
}

7.1 Deadlock Avoidance

Deadlocks occur when threads acquire multiple locks in different orders. Always acquire locks in a consistent order:

use std::sync::{Arc, Mutex};
use std::thread;

fn transfer(from: &Mutex<i64>, to: &Mutex<i64>, amount: i64) {
    // BUG: if two threads call transfer(a, b) and transfer(b, a)
    // simultaneously, they can deadlock.
    // FIX: always lock in a canonical order (e.g., by pointer address)
    let (first, second) = if (from as *const _) < (to as *const _) {
        (from, to)
    } else {
        (to, from)
    };

    let mut f = first.lock().unwrap();
    let mut t = second.lock().unwrap();
    *f -= amount;
    *t += amount;
}

fn main() {
    let a = Arc::new(Mutex::new(100i64));
    let b = Arc::new(Mutex::new(200i64));

    let (a1, b1) = (Arc::clone(&a), Arc::clone(&b));
    let (a2, b2) = (Arc::clone(&a), Arc::clone(&b));

    let h1 = thread::spawn(move || transfer(&a1, &b1, 30));
    let h2 = thread::spawn(move || transfer(&b2, &a2, 50));

    h1.join().unwrap();
    h2.join().unwrap();

    println!("a = {}", a.lock().unwrap()); // 120
    println!("b = {}", b.lock().unwrap()); // 180
}

8. Data Parallelism with Rayon

Rayon provides parallel iterators that automatically distribute work across a thread pool. Add it with cargo add rayon.

// Cargo.toml: rayon = "1"
use rayon::prelude::*;

fn is_prime(n: u64) -> bool {
    if n < 2 { return false; }
    if n == 2 { return true; }
    if n % 2 == 0 { return false; }
    let mut i = 3u64;
    while i * i <= n {
        if n % i == 0 { return false; }
        i += 2;
    }
    true
}

fn main() {
    let numbers: Vec<u64> = (2..100_000).collect();

    // par_iter() distributes work across Rayon's thread pool
    let primes: Vec<u64> = numbers
        .par_iter()
        .copied()
        .filter(|&n| is_prime(n))
        .collect();

    println!("found {} primes below 100,000", primes.len()); // 9592

    // parallel map-reduce
    let sum: u64 = (1u64..=1_000_000)
        .into_par_iter()
        .map(|x| x * x)
        .sum();
    println!("sum of squares 1..1M = {sum}");
}

Rayon's par_iter() is a drop-in replacement for iter(). It uses a work-stealing scheduler to balance load automatically — ideal for CPU-bound tasks.

8.1 Parallel Sort and Custom Pool

use rayon::prelude::*;
use rayon::ThreadPoolBuilder;

fn main() {
    // parallel sort (much faster than sequential for large slices)
    let mut data: Vec<i32> = (0..1_000_000).rev().collect();
    data.par_sort();
    println!("sorted: first={}, last={}", data[0], data[data.len()-1]);

    // custom thread pool with 4 threads
    let pool = ThreadPoolBuilder::new()
        .num_threads(4)
        .build()
        .unwrap();

    let result: i32 = pool.install(|| {
        (1..=100).into_par_iter().sum()
    });
    println!("sum 1..100 = {result}"); // 5050
}

9. Send and Sync — The Type System Enforces Thread Safety

Send means ownership of a value can be transferred to another thread. Sync means a shared reference &T can be sent to another thread. These are auto-traits — the compiler implements them automatically for types that qualify.

use std::sync::{Arc, Mutex};
use std::cell::RefCell;
use std::rc::Rc;

fn requires_send<T: Send>(_: T) {}
fn requires_sync<T: Sync>(_: &T) {}

fn main() {
    // i32, String, Vec — Send + Sync ✓
    requires_send(42i32);
    requires_sync(&String::from("hello"));

    // Rc is NOT Send (not thread-safe reference counting)
    // requires_send(Rc::new(1)); // compile error!

    // RefCell is NOT Sync (dynamic borrow checking is not thread-safe)
    // requires_sync(&RefCell::new(1)); // compile error!

    // Arc IS Send + Sync (atomic reference counting)
    requires_send(Arc::new(1));

    // Arc<Mutex<T>> IS Send + Sync (mutex provides synchronization)
    let shared = Arc::new(Mutex::new(vec![1, 2, 3]));
    requires_send(Arc::clone(&shared));
    requires_sync(&shared);
}

9.1 Manually Implementing Send/Sync

If you have a type that is logically thread-safe but the compiler cannot verify it automatically (e.g., a wrapper around a raw pointer), you can implement Send and Sync manually using unsafe.

use std::ptr::NonNull;

// A thread-safe wrapper around a raw pointer
struct SharedBuffer {
    ptr: NonNull<u8>,
    len: usize,
}

// SAFETY: We guarantee no aliased mutable access across threads
unsafe impl Send for SharedBuffer {}
unsafe impl Sync for SharedBuffer {}

impl SharedBuffer {
    fn new(len: usize) -> Self {
        let ptr = unsafe {
            let layout = std::alloc::Layout::from_size_align(len, 1).unwrap();
            NonNull::new(std::alloc::alloc_zeroed(layout)).unwrap()
        };
        SharedBuffer { ptr, len }
    }

    fn as_slice(&self) -> &[u8] {
        unsafe { std::slice::from_raw_parts(self.ptr.as_ptr(), self.len) }
    }
}

10. Practical Patterns

10.1 Thread Pool with Work Queue

use std::sync::{Arc, Mutex};
use std::thread;

type Job = Box<dyn FnOnce() + Send + 'static>;

struct ThreadPool {
    workers: Vec<thread::JoinHandle<()>>,
    sender: std::sync::mpsc::SyncSender<Option<Job>>,
}

impl ThreadPool {
    fn new(size: usize) -> Self {
        let (tx, rx) = std::sync::mpsc::sync_channel::<Option<Job>>(64);
        let rx = Arc::new(Mutex::new(rx));
        let mut workers = Vec::with_capacity(size);

        for id in 0..size {
            let rx = Arc::clone(&rx);
            workers.push(thread::Builder::new()
                .name(format!("worker-{id}"))
                .spawn(move || loop {
                    let job = rx.lock().unwrap().recv().unwrap();
                    match job {
                        Some(f) => f(),
                        None    => break, // shutdown signal
                    }
                })
                .unwrap());
        }

        ThreadPool { workers, sender: tx }
    }

    fn execute(&self, job: impl FnOnce() + Send + 'static) {
        self.sender.send(Some(Box::new(job))).unwrap();
    }

    fn shutdown(self) {
        for _ in &self.workers {
            self.sender.send(None).unwrap(); // one None per worker
        }
        for w in self.workers {
            w.join().unwrap();
        }
    }
}

fn main() {
    let pool = ThreadPool::new(4);

    for i in 0..12 {
        pool.execute(move || {
            println!("job {i} on {:?}", thread::current().name());
        });
    }

    pool.shutdown();
}

10.2 Parallel Map Pattern

use std::thread;

fn parallel_map<T, U, F>(data: Vec<T>, f: F) -> Vec<U>
where
    T: Send + 'static,
    U: Send + 'static,
    F: Fn(T) -> U + Send + Sync + 'static,
{
    let f = std::sync::Arc::new(f);
    let handles: Vec<_> = data
        .into_iter()
        .map(|item| {
            let f = std::sync::Arc::clone(&f);
            thread::spawn(move || f(item))
        })
        .collect();

    handles.into_iter().map(|h| h.join().unwrap()).collect()
}

fn main() {
    let results = parallel_map(vec![1u64, 2, 3, 4, 5], |n| {
        // simulate expensive computation
        n * n
    });
    println!("{results:?}"); // [1, 4, 9, 16, 25]
}
