Channels & Message Passing in Rust

Rust Channels Concurrency

1. std::sync::mpsc — Multiple Producer, Single Consumer

The standard library provides mpsc::channel() (unbounded) and mpsc::sync_channel() (bounded). "mpsc" stands for multiple producer, single consumer — you can clone the Sender but there is only one Receiver.

use std::sync::mpsc;
use std::thread;

fn main() {
    // unbounded channel — sender never blocks
    let (tx, rx) = mpsc::channel::<String>();

    let handle = thread::spawn(move || {
        tx.send("hello".to_string()).unwrap();
        tx.send("world".to_string()).unwrap();
        // tx is dropped here; rx.recv() will return Err when all senders drop
    });

    // recv() blocks until a message arrives (or all senders drop)
    while let Ok(msg) = rx.recv() {
        println!("received: {msg}");
    }
    println!("channel closed");

    handle.join().unwrap();
}
// Output:
// received: hello
// received: world
// channel closed

1.1 try_recv and recv_timeout

use std::sync::mpsc;
use std::time::Duration;
use std::thread;

fn main() {
    let (tx, rx) = mpsc::channel::<i32>();

    thread::spawn(move || {
        thread::sleep(Duration::from_millis(50));
        tx.send(42).unwrap();
    });

    // try_recv is non-blocking
    match rx.try_recv() {
        Ok(v)                              => println!("immediate: {v}"),
        Err(mpsc::TryRecvError::Empty)     => println!("no message yet"),
        Err(mpsc::TryRecvError::Disconnected) => println!("disconnected"),
    }

    // recv_timeout blocks for at most the given duration
    match rx.recv_timeout(Duration::from_millis(200)) {
        Ok(v)                                 => println!("got: {v}"),
        Err(mpsc::RecvTimeoutError::Timeout)  => println!("timed out"),
        Err(mpsc::RecvTimeoutError::Disconnected) => println!("disconnected"),
    }
}
// Output:
// no message yet
// got: 42

2. Multiple Producers

Clone the Sender to create multiple producers. All clones share the same channel; each message is delivered to the single Receiver.

use std::sync::mpsc;
use std::thread;

#[derive(Debug)]
struct WorkResult {
    worker_id: usize,
    value: u64,
}

fn main() {
    let (tx, rx) = mpsc::channel::<WorkResult>();
    let mut handles = Vec::new();

    for id in 0..4 {
        let tx = tx.clone(); // clone the sender for each thread
        handles.push(thread::spawn(move || {
            // simulate work
            let result = (id as u64 + 1) * 100;
            tx.send(WorkResult { worker_id: id, value: result }).unwrap();
        }));
    }

    // Drop the original tx so the channel closes when all workers finish
    drop(tx);

    // Collect all results
    let mut results: Vec<WorkResult> = rx.iter().collect(); // iter() = loop recv()
    results.sort_by_key(|r| r.worker_id);

    for r in &results {
        println!("worker {}: {}", r.worker_id, r.value);
    }

    for h in handles { h.join().unwrap(); }
}

Key pattern: always drop(tx) after cloning. If the original sender is not dropped, rx.iter() never terminates because the channel looks open.

3. Bounded Channels — sync_channel

sync_channel(n) creates a channel with a fixed-size buffer. The sender blocks when the buffer is full, providing back-pressure automatically.

use std::sync::mpsc;
use std::thread;
use std::time::Duration;

fn main() {
    // buffer capacity = 2; sender blocks on the 3rd send until receiver reads
    let (tx, rx) = mpsc::sync_channel::<i32>(2);

    let producer = thread::spawn(move || {
        for i in 0..5 {
            println!("sending {i}...");
            tx.send(i).unwrap(); // blocks when buffer full
            println!("sent {i}");
        }
    });

    thread::sleep(Duration::from_millis(50)); // let producer fill the buffer

    for _ in 0..5 {
        thread::sleep(Duration::from_millis(20));
        let v = rx.recv().unwrap();
        println!("received {v}");
    }

    producer.join().unwrap();
}
// The producer will block after sending 2 messages and
// resume as the consumer reads them — natural back-pressure.

3.1 Rendezvous Channel (capacity = 0)

use std::sync::mpsc;
use std::thread;

fn main() {
    // capacity 0 = rendezvous: send blocks until recv is ready
    let (tx, rx) = mpsc::sync_channel::<i32>(0);

    let sender = thread::spawn(move || {
        println!("sender: waiting for receiver...");
        tx.send(99).unwrap(); // blocks until receiver calls recv()
        println!("sender: handed off");
    });

    thread::sleep(std::time::Duration::from_millis(50));
    println!("receiver: calling recv()");
    let v = rx.recv().unwrap();
    println!("receiver: got {v}");

    sender.join().unwrap();
}

4. Sending Complex Types and Errors

Anything that implements Send can travel through a channel. Use enums to multiplex different message kinds.

use std::sync::mpsc;
use std::thread;

// A typed command enum — clean alternative to trait objects
#[derive(Debug)]
enum Command {
    Process(Vec<u8>),
    SetConfig { key: String, value: String },
    Shutdown,
}

fn main() {
    let (tx, rx) = mpsc::channel::<Command>();

    let worker = thread::spawn(move || {
        loop {
            match rx.recv().unwrap() {
                Command::Process(data) => {
                    println!("processing {} bytes", data.len());
                }
                Command::SetConfig { key, value } => {
                    println!("config: {key} = {value}");
                }
                Command::Shutdown => {
                    println!("shutting down");
                    break;
                }
            }
        }
    });

    tx.send(Command::SetConfig {
        key: "timeout".to_string(),
        value: "30s".to_string(),
    }).unwrap();
    tx.send(Command::Process(vec![1, 2, 3, 4])).unwrap();
    tx.send(Command::Shutdown).unwrap();

    worker.join().unwrap();
}

4.1 Result over Channels

use std::sync::mpsc;
use std::thread;
use std::num::ParseIntError;

fn parse_worker(input: &str, tx: mpsc::Sender<Result<i32, ParseIntError>>) {
    let result = input.trim().parse::<i32>();
    tx.send(result).unwrap();
}

fn main() {
    let (tx, rx) = mpsc::channel();

    let inputs = vec!["42", "not_a_number", "100"];
    let mut handles = Vec::new();

    for input in inputs {
        let tx = tx.clone();
        let input = input.to_string();
        handles.push(thread::spawn(move || parse_worker(&input, tx)));
    }
    drop(tx);

    let (ok, err): (Vec<_>, Vec<_>) = rx.iter().partition(Result::is_ok);
    println!("parsed: {:?}", ok.into_iter().map(Result::unwrap).collect::<Vec<_>>());
    println!("errors: {}", err.len());

    for h in handles { h.join().unwrap(); }
}

5. The Actor Pattern

An actor is a thread that owns exclusive mutable state and communicates only through message passing. This pattern eliminates shared state and makes concurrency bugs much harder to introduce.

use std::sync::mpsc;
use std::thread;
use std::collections::HashMap;

// Messages the actor understands
enum CacheMsg {
    Set(String, String),
    Get(String, mpsc::Sender<Option<String>>),
    Delete(String),
    Stats(mpsc::Sender<usize>),
}

// Actor handle — cheap to clone, hides implementation
#[derive(Clone)]
struct CacheHandle(mpsc::Sender<CacheMsg>);

impl CacheHandle {
    fn new() -> Self {
        let (tx, rx) = mpsc::channel::<CacheMsg>();

        thread::spawn(move || {
            let mut store: HashMap<String, String> = HashMap::new();

            loop {
                match rx.recv() {
                    Ok(CacheMsg::Set(k, v)) => { store.insert(k, v); }
                    Ok(CacheMsg::Get(k, reply)) => {
                        reply.send(store.get(&k).cloned()).unwrap();
                    }
                    Ok(CacheMsg::Delete(k)) => { store.remove(&k); }
                    Ok(CacheMsg::Stats(reply)) => {
                        reply.send(store.len()).unwrap();
                    }
                    Err(_) => break, // all senders dropped
                }
            }
        });

        CacheHandle(tx)
    }

    fn set(&self, key: impl Into<String>, val: impl Into<String>) {
        self.0.send(CacheMsg::Set(key.into(), val.into())).unwrap();
    }

    fn get(&self, key: impl Into<String>) -> Option<String> {
        let (tx, rx) = mpsc::channel();
        self.0.send(CacheMsg::Get(key.into(), tx)).unwrap();
        rx.recv().unwrap()
    }

    fn stats(&self) -> usize {
        let (tx, rx) = mpsc::channel();
        self.0.send(CacheMsg::Stats(tx)).unwrap();
        rx.recv().unwrap()
    }
}

fn main() {
    let cache = CacheHandle::new();
    let cache2 = cache.clone(); // give a copy to another thread

    let h = thread::spawn(move || {
        cache2.set("key1", "value1");
        cache2.set("key2", "value2");
    });
    h.join().unwrap();

    cache.set("key3", "value3");
    println!("key1 = {:?}", cache.get("key1")); // Some("value1")
    println!("key9 = {:?}", cache.get("key9")); // None
    println!("size = {}", cache.stats());         // 3
}

6. Pipeline Pattern

Chain threads together with channels to create processing pipelines. Each stage reads from one channel and writes to another.

use std::sync::mpsc;
use std::thread;

// Stage 1: generate numbers
fn generate(tx: mpsc::SyncSender<u64>) {
    for n in 0u64..20 {
        tx.send(n).unwrap();
    }
}

// Stage 2: square numbers
fn square(rx: mpsc::Receiver<u64>, tx: mpsc::SyncSender<u64>) {
    for n in rx {
        tx.send(n * n).unwrap();
    }
}

// Stage 3: filter even numbers
fn filter_even(rx: mpsc::Receiver<u64>, tx: mpsc::SyncSender<u64>) {
    for n in rx {
        if n % 2 == 0 {
            tx.send(n).unwrap();
        }
    }
}

fn main() {
    let (tx1, rx1) = mpsc::sync_channel::<u64>(8);
    let (tx2, rx2) = mpsc::sync_channel::<u64>(8);
    let (tx3, rx3) = mpsc::sync_channel::<u64>(8);

    let h1 = thread::spawn(move || generate(tx1));
    let h2 = thread::spawn(move || square(rx1, tx2));
    let h3 = thread::spawn(move || filter_even(rx2, tx3));

    // collect results
    let results: Vec<u64> = rx3.iter().collect();
    println!("{results:?}");
    // [0, 4, 16, 36, 64, 100, 144, 196, 256, 324]

    h1.join().unwrap();
    h2.join().unwrap();
    h3.join().unwrap();
}

7. crossbeam-channel — Advanced Primitives

crossbeam-channel extends mpsc with multiple consumers, select! over multiple channels, and after/tick senders. Add it with cargo add crossbeam-channel.

// Cargo.toml: crossbeam-channel = "0.5"
use crossbeam_channel::{bounded, select, tick, after};
use std::time::Duration;
use std::thread;

fn main() {
    let (tx, rx) = bounded::<String>(8); // MPMC: both ends can be cloned!

    // Multiple consumers
    for id in 0..3 {
        let rx = rx.clone();
        thread::spawn(move || {
            for msg in &rx {
                println!("consumer {id}: {msg}");
            }
        });
    }

    // Producers
    let tx2 = tx.clone();
    thread::spawn(move || {
        for i in 0..6 {
            tx2.send(format!("msg-{i}")).unwrap();
        }
    });

    // Original sender also sends
    for i in 6..9 {
        tx.send(format!("msg-{i}")).unwrap();
    }
    drop(tx); // channel closes when all senders drop

    thread::sleep(Duration::from_millis(100));
}

7.1 select! — Wait on Multiple Channels

use crossbeam_channel::{bounded, tick, after, select};
use std::time::Duration;
use std::thread;

fn main() {
    let (data_tx, data_rx) = bounded::<i32>(8);
    let ticker = tick(Duration::from_millis(200));       // fires every 200ms
    let deadline = after(Duration::from_millis(1_000));  // fires once after 1s

    thread::spawn(move || {
        for i in 0..5 {
            thread::sleep(Duration::from_millis(150));
            data_tx.send(i).unwrap();
        }
    });

    loop {
        select! {
            recv(data_rx) -> msg => {
                match msg {
                    Ok(v) => println!("data: {v}"),
                    Err(_) => println!("data channel closed"),
                }
            }
            recv(ticker) -> _ => {
                println!("tick");
            }
            recv(deadline) -> _ => {
                println!("deadline reached, exiting");
                break;
            }
        }
    }
}

7.2 Unbounded Channel Performance

use crossbeam_channel::unbounded;
use std::thread;

fn main() {
    let (tx, rx) = unbounded::<u64>();

    // crossbeam unbounded is significantly faster than std::mpsc for
    // high-throughput scenarios (lock-free FIFO queue internally)
    let producer = thread::spawn(move || {
        for i in 0..1_000_000u64 {
            tx.send(i).unwrap();
        }
    });

    let consumer = thread::spawn(move || {
        let mut sum = 0u64;
        for v in &rx {
            sum += v;
        }
        sum
    });

    producer.join().unwrap();
    let total = consumer.join().unwrap();
    println!("sum = {total}"); // 499999500000
}

8. Channels vs Mutex — When to Use Which

Scenario Prefer Why
Pipeline / streaming data Channel Natural ownership transfer
Actor / encapsulated state Channel No shared mutable state
Shared counter / cache Mutex Simpler, lower overhead
Reader-heavy shared data RwLock Concurrent reads
Single integer counter Atomic Lock-free, fastest
Select over multiple events crossbeam select! No equivalent with Mutex 
// Rule of thumb: if data flows in one direction → channel
//                if data is shared and mutated in place → Mutex/RwLock

use std::sync::{Arc, Mutex, mpsc};
use std::thread;

// Good use of Mutex: shared configuration read/written by any thread
fn mutex_example() {
    let config = Arc::new(Mutex::new(std::collections::HashMap::<String, String>::new()));

    let c1 = Arc::clone(&config);
    thread::spawn(move || {
        c1.lock().unwrap().insert("key".to_string(), "value".to_string());
    }).join().unwrap();

    println!("{:?}", config.lock().unwrap().get("key")); // Some("value")
}

// Good use of channel: log messages flow from workers to a collector
fn channel_example() {
    let (tx, rx) = mpsc::channel::<String>();

    for i in 0..4 {
        let tx = tx.clone();
        thread::spawn(move || {
            tx.send(format!("worker {i} done")).unwrap();
        });
    }
    drop(tx);

    for msg in rx {
        println!("{msg}");
    }
}

fn main() {
    mutex_example();
    channel_example();
}
