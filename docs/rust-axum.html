2. Routing and Handlers

Axum handlers are async functions whose parameters are extractors — the framework calls them with data pulled from the request automatically.

use axum::{
    routing::{get, post, put, delete},
    Router, Json, Path, Query, State,
    extract::rejection::JsonRejection,
    http::StatusCode,
    response::{IntoResponse, Response},
};
use serde::{Deserialize, Serialize};
use std::sync::Arc;

#[derive(Debug, Serialize, Deserialize, Clone)]
struct User {
    id: u64,
    name: String,
    email: String,
}

// Simple GET handler — returns plain text
async fn health() -> &'static str {
    "OK"
}

// Path extractor: /users/:id
async fn get_user(Path(id): Path<u64>) -> impl IntoResponse {
    let user = User { id, name: "Alice".to_string(), email: "alice@example.com".to_string() };
    (StatusCode::OK, Json(user))
}

// Query extractor: /users?page=2&per_page=10
#[derive(Deserialize)]
struct Pagination {
    #[serde(default = "default_page")]
    page: u32,
    #[serde(default = "default_per_page")]
    per_page: u32,
}
fn default_page() -> u32 { 1 }
fn default_per_page() -> u32 { 20 }

async fn list_users(Query(pag): Query<Pagination>) -> impl IntoResponse {
    Json(serde_json::json!({
        "page": pag.page,
        "per_page": pag.per_page,
        "data": [],
    }))
}

// JSON body extractor
#[derive(Deserialize)]
struct CreateUser {
    name: String,
    email: String,
}

async fn create_user(Json(body): Json<CreateUser>) -> impl IntoResponse {
    let user = User { id: 1, name: body.name, email: body.email };
    (StatusCode::CREATED, Json(user))
}

#[tokio::main]
async fn main() {
    let app = Router::new()
        .route("/health",      get(health))
        .route("/users",       get(list_users).post(create_user))
        .route("/users/:id",   get(get_user));

    let listener = tokio::net::TcpListener::bind("0.0.0.0:3000").await.unwrap();
    println!("listening on {}", listener.local_addr().unwrap());
    axum::serve(listener, app).await.unwrap();
}

3. Shared Application State

Use State to share data across handlers — database pools, caches, configuration.

use axum::{extract::State, routing::get, Router, Json};
use std::sync::{Arc, RwLock};
use std::collections::HashMap;

// Application state — wrap in Arc for cheap cloning across tasks
#[derive(Clone)]
struct AppState {
    db: Arc<RwLock<HashMap<u64, User>>>,
    config: Arc<Config>,
}

#[derive(Debug)]
struct Config {
    max_users: usize,
    app_name: String,
}

// Handlers receive State<AppState> as an extractor
async fn list_users_state(State(state): State<AppState>) -> Json<Vec<User>> {
    let db = state.db.read().unwrap();
    Json(db.values().cloned().collect())
}

async fn create_user_state(
    State(state): State<AppState>,
    Json(body): Json<CreateUser>,
) -> impl IntoResponse {
    let mut db = state.db.write().unwrap();

    if db.len() >= state.config.max_users {
        return (StatusCode::FORBIDDEN, Json(serde_json::json!({
            "error": "user limit reached"
        }))).into_response();
    }

    let id = db.len() as u64 + 1;
    let user = User { id, name: body.name, email: body.email };
    db.insert(id, user.clone());
    (StatusCode::CREATED, Json(user)).into_response()
}

#[tokio::main]
async fn main() {
    let state = AppState {
        db: Arc::new(RwLock::new(HashMap::new())),
        config: Arc::new(Config {
            max_users: 1000,
            app_name: "my-api".to_string(),
        }),
    };

    let app = Router::new()
        .route("/users", get(list_users_state).post(create_user_state))
        .with_state(state);

    let listener = tokio::net::TcpListener::bind("0.0.0.0:3000").await.unwrap();
    axum::serve(listener, app).await.unwrap();
}

4. Error Handling

Define a custom error type that implements IntoResponse. This lets handlers return Result<T, AppError> cleanly.

use axum::{http::StatusCode, response::{IntoResponse, Response}, Json};
use serde_json::json;

#[derive(Debug)]
enum AppError {
    NotFound(String),
    Conflict(String),
    BadRequest(String),
    Internal(anyhow::Error),
}

impl IntoResponse for AppError {
    fn into_response(self) -> Response {
        let (status, message) = match self {
            AppError::NotFound(m)  => (StatusCode::NOT_FOUND, m),
            AppError::Conflict(m)  => (StatusCode::CONFLICT, m),
            AppError::BadRequest(m) => (StatusCode::BAD_REQUEST, m),
            AppError::Internal(e)  => {
                tracing::error!("internal error: {e:#}");
                (StatusCode::INTERNAL_SERVER_ERROR, "internal server error".to_string())
            }
        };
        (status, Json(json!({ "error": message }))).into_response()
    }
}

// Propagate anyhow errors easily
impl<E: Into<anyhow::Error>> From<E> for AppError {
    fn from(e: E) -> Self { AppError::Internal(e.into()) }
}

// Handlers can now use the ? operator
async fn get_user_or_404(
    Path(id): Path<u64>,
    State(state): State<AppState>,
) -> Result<Json<User>, AppError> {
    let db = state.db.read().unwrap();
    db.get(&id)
        .cloned()
        .map(Json)
        .ok_or_else(|| AppError::NotFound(format!("user {id} not found")))
}

async fn delete_user(
    Path(id): Path<u64>,
    State(state): State<AppState>,
) -> Result<StatusCode, AppError> {
    let mut db = state.db.write().unwrap();
    if db.remove(&id).is_some() {
        Ok(StatusCode::NO_CONTENT)
    } else {
        Err(AppError::NotFound(format!("user {id} not found")))
    }
}

5. Middleware with Tower

Axum uses the Tower ecosystem for middleware. tower-http provides ready-made middleware for common concerns.

use axum::Router;
use tower_http::{
    cors::{CorsLayer, Any},
    trace::TraceLayer,
    compression::CompressionLayer,
    timeout::TimeoutLayer,
    limit::RequestBodyLimitLayer,
};
use tower::ServiceBuilder;
use std::time::Duration;
use http::Method;

fn build_app(state: AppState) -> Router {
    Router::new()
        .route("/users", get(list_users_state).post(create_user_state))
        .route("/users/:id", get(get_user_or_404).delete(delete_user))
        .with_state(state)
        .layer(
            ServiceBuilder::new()
                // Request timeout
                .layer(TimeoutLayer::new(Duration::from_secs(30)))
                // Limit request body to 1 MiB
                .layer(RequestBodyLimitLayer::new(1024 * 1024))
                // Gzip response compression
                .layer(CompressionLayer::new())
                // Structured request/response tracing
                .layer(TraceLayer::new_for_http())
                // CORS
                .layer(
                    CorsLayer::new()
                        .allow_origin(Any)
                        .allow_methods([Method::GET, Method::POST, Method::PUT, Method::DELETE])
                        .allow_headers(Any),
                ),
        )
}

5.1 Custom Middleware

use axum::{
    middleware::{self, Next},
    extract::Request,
    response::Response,
    http::HeaderMap,
};

// Authentication middleware
async fn require_api_key(
    headers: HeaderMap,
    request: Request,
    next: Next,
) -> Result<Response, StatusCode> {
    let key = headers
        .get("x-api-key")
        .and_then(|v| v.to_str().ok());

    match key {
        Some("secret-key") => Ok(next.run(request).await),
        _ => Err(StatusCode::UNAUTHORIZED),
    }
}

// Request ID middleware
async fn add_request_id(
    mut request: Request,
    next: Next,
) -> Response {
    let id = uuid_like(); // e.g. from ulid or uuid crate
    request.extensions_mut().insert(RequestId(id.clone()));
    let mut response = next.run(request).await;
    response.headers_mut().insert(
        "x-request-id",
        id.parse().unwrap(),
    );
    response
}

fn uuid_like() -> String {
    use std::time::UNIX_EPOCH;
    format!("{:x}", std::time::SystemTime::now()
        .duration_since(UNIX_EPOCH).unwrap().as_nanos())
}

struct RequestId(String);

// Apply middleware to specific routes only
fn protected_routes(state: AppState) -> Router {
    Router::new()
        .route("/admin/stats", get(admin_stats))
        .layer(middleware::from_fn(require_api_key))
        .with_state(state)
}

async fn admin_stats() -> Json<serde_json::Value> {
    Json(json!({ "total_users": 42 }))
}

6. Custom Extractors

use axum::{
    async_trait,
    extract::{FromRequest, FromRequestParts},
    http::{request::Parts, Request, HeaderMap},
};

// Extract and validate a Bearer token from the Authorization header
struct AuthUser {
    user_id: u64,
    email: String,
}

#[async_trait]
impl<S: Send + Sync> FromRequestParts<S> for AuthUser {
    type Rejection = (StatusCode, Json<serde_json::Value>);

    async fn from_request_parts(parts: &mut Parts, _state: &S) -> Result<Self, Self::Rejection> {
        let token = parts.headers
            .get("authorization")
            .and_then(|v| v.to_str().ok())
            .and_then(|v| v.strip_prefix("Bearer "))
            .ok_or_else(|| (
                StatusCode::UNAUTHORIZED,
                Json(json!({ "error": "missing or invalid Authorization header" })),
            ))?;

        // In real code: validate JWT, look up session, etc.
        if token == "valid-token" {
            Ok(AuthUser { user_id: 1, email: "user@example.com".to_string() })
        } else {
            Err((StatusCode::UNAUTHORIZED, Json(json!({ "error": "invalid token" }))))
        }
    }
}

// Use in handlers — AuthUser is extracted automatically
async fn my_profile(auth: AuthUser) -> Json<serde_json::Value> {
    Json(json!({ "user_id": auth.user_id, "email": auth.email }))
}

7. File Upload and Multipart

use axum::extract::Multipart;

async fn upload_file(mut multipart: Multipart) -> Result<Json<serde_json::Value>, AppError> {
    let mut uploaded = Vec::new();

    while let Some(field) = multipart.next_field().await? {
        let name = field.name().unwrap_or("unnamed").to_string();
        let file_name = field.file_name().map(String::from);
        let content_type = field.content_type().map(String::from);
        let data = field.bytes().await?;

        println!("field: {name}, filename: {file_name:?}, size: {}", data.len());

        // In real code: save to disk or object storage
        uploaded.push(json!({
            "field": name,
            "filename": file_name,
            "content_type": content_type,
            "size": data.len(),
        }));
    }

    Ok(Json(json!({ "uploaded": uploaded })))
}

8. Complete REST API Example

use axum::{routing::{get, post, put, delete}, Router, Json, Path, State, extract::Query};
use serde::{Deserialize, Serialize};
use std::sync::{Arc, RwLock};
use std::collections::HashMap;

#[derive(Debug, Serialize, Deserialize, Clone)]
struct Todo {
    id: u64,
    title: String,
    done: bool,
}

#[derive(Deserialize)]
struct CreateTodo { title: String }

#[derive(Deserialize)]
struct UpdateTodo {
    title: Option<String>,
    done: Option<bool>,
}

#[derive(Deserialize)]
struct TodoFilter {
    done: Option<bool>,
}

type Db = Arc<RwLock<HashMap<u64, Todo>>>;

async fn list_todos(State(db): State<Db>, Query(f): Query<TodoFilter>) -> Json<Vec<Todo>> {
    let db = db.read().unwrap();
    let mut todos: Vec<Todo> = db.values()
        .filter(|t| f.done.is_none() || f.done == Some(t.done))
        .cloned()
        .collect();
    todos.sort_by_key(|t| t.id);
    Json(todos)
}

async fn create_todo(State(db): State<Db>, Json(body): Json<CreateTodo>) -> impl IntoResponse {
    let mut db = db.write().unwrap();
    let id = db.len() as u64 + 1;
    let todo = Todo { id, title: body.title, done: false };
    db.insert(id, todo.clone());
    (StatusCode::CREATED, Json(todo))
}

async fn get_todo(State(db): State<Db>, Path(id): Path<u64>) -> Result<Json<Todo>, StatusCode> {
    db.read().unwrap().get(&id).cloned().map(Json).ok_or(StatusCode::NOT_FOUND)
}

async fn update_todo(
    State(db): State<Db>,
    Path(id): Path<u64>,
    Json(body): Json<UpdateTodo>,
) -> Result<Json<Todo>, StatusCode> {
    let mut db = db.write().unwrap();
    let todo = db.get_mut(&id).ok_or(StatusCode::NOT_FOUND)?;
    if let Some(title) = body.title { todo.title = title; }
    if let Some(done)  = body.done  { todo.done = done; }
    Ok(Json(todo.clone()))
}

async fn delete_todo(State(db): State<Db>, Path(id): Path<u64>) -> StatusCode {
    if db.write().unwrap().remove(&id).is_some() { StatusCode::NO_CONTENT }
    else { StatusCode::NOT_FOUND }
}

#[tokio::main]
async fn main() {
    tracing_subscriber::fmt::init();

    let db: Db = Arc::new(RwLock::new(HashMap::new()));

    let app = Router::new()
        .route("/todos",     get(list_todos).post(create_todo))
        .route("/todos/:id", get(get_todo).put(update_todo).delete(delete_todo))
        .with_state(db)
        .layer(tower_http::trace::TraceLayer::new_for_http());

    let listener = tokio::net::TcpListener::bind("0.0.0.0:3000").await.unwrap();
    tracing::info!("listening on {}", listener.local_addr().unwrap());
    axum::serve(listener, app).await.unwrap();
}

9. Testing Axum Handlers

#[cfg(test)]
mod tests {
    use super::*;
    use axum::body::Body;
    use axum::http::{Request, StatusCode};
    use tower::ServiceExt; // for .oneshot()

    fn test_app() -> Router {
        let db: Db = Arc::new(RwLock::new(HashMap::new()));
        Router::new()
            .route("/todos",     get(list_todos).post(create_todo))
            .route("/todos/:id", get(get_todo).put(update_todo).delete(delete_todo))
            .with_state(db)
    }

    #[tokio::test]
    async fn test_list_empty() {
        let app = test_app();
        let response = app
            .oneshot(Request::builder().uri("/todos").body(Body::empty()).unwrap())
            .await
            .unwrap();

        assert_eq!(response.status(), StatusCode::OK);
        let body = axum::body::to_bytes(response.into_body(), 1024).await.unwrap();
        let json: serde_json::Value = serde_json::from_slice(&body).unwrap();
        assert_eq!(json, serde_json::json!([]));
    }

    #[tokio::test]
    async fn test_create_and_fetch() {
        let app = test_app();

        // Create
        let create_resp = app.clone()
            .oneshot(
                Request::builder()
                    .method("POST")
                    .uri("/todos")
                    .header("content-type", "application/json")
                    .body(Body::from(r#"{"title":"test todo"}"#))
                    .unwrap(),
            )
            .await
            .unwrap();

        assert_eq!(create_resp.status(), StatusCode::CREATED);
        let body = axum::body::to_bytes(create_resp.into_body(), 4096).await.unwrap();
        let todo: serde_json::Value = serde_json::from_slice(&body).unwrap();
        assert_eq!(todo["title"], "test todo");
        assert_eq!(todo["done"], false);
    }

    #[tokio::test]
    async fn test_not_found() {
        let app = test_app();
        let resp = app
            .oneshot(Request::builder().uri("/todos/999").body(Body::empty()).unwrap())
            .await
            .unwrap();
        assert_eq!(resp.status(), StatusCode::NOT_FOUND);
    }
}
