Design Patterns & Idioms in Rust

Rust Patterns Idioms

1. Newtype Pattern

Wrap a primitive in a single-field tuple struct to create a distinct type. This prevents mixing up values that have the same underlying type but different semantics.

// Without newtype: easy to mix up Meters and Kilograms at call sites
fn speed(distance: f64, time: f64) -> f64 { distance / time }

// With newtype: the type system catches mistakes
#[derive(Debug, Clone, Copy, PartialEq, PartialOrd)]
struct Meters(f64);

#[derive(Debug, Clone, Copy, PartialEq, PartialOrd)]
struct Seconds(f64);

#[derive(Debug, Clone, Copy)]
struct MetersPerSecond(f64);

fn typed_speed(distance: Meters, time: Seconds) -> MetersPerSecond {
    MetersPerSecond(distance.0 / time.0)
}

fn main() {
    let d = Meters(100.0);
    let t = Seconds(9.58);
    let v = typed_speed(d, t);
    println!("{:.2} m/s", v.0); // 10.44 m/s

    // typed_speed(t, d); // compile error: wrong argument order!
}

1.1 Newtype with Deref for Ergonomics

use std::ops::{Deref, DerefMut};

/// A non-empty Vec — the invariant is enforced by the constructor.
pub struct NonEmptyVec<T>(Vec<T>);

impl<T> NonEmptyVec<T> {
    pub fn new(first: T) -> Self {
        NonEmptyVec(vec![first])
    }

    pub fn from_vec(v: Vec<T>) -> Option<Self> {
        if v.is_empty() { None } else { Some(NonEmptyVec(v)) }
    }

    pub fn first(&self) -> &T { &self.0[0] } // always safe
}

// Deref lets NonEmptyVec<T> be used wherever &Vec<T> or &[T> is expected
impl<T> Deref for NonEmptyVec<T> {
    type Target = Vec<T>;
    fn deref(&self) -> &Self::Target { &self.0 }
}

impl<T> DerefMut for NonEmptyVec<T> {
    fn deref_mut(&mut self) -> &mut Self::Target { &mut self.0 }
}

fn main() {
    let mut v = NonEmptyVec::new(1i32);
    v.push(2); // works via Deref to Vec
    v.push(3);
    println!("first: {}", v.first()); // 1
    println!("len: {}", v.len());     // 3 — Vec method via Deref
}

2. Typestate Pattern

Encode the state of an object in its type. Invalid state transitions become compile errors — impossible to call send() before connect(), for example.

use std::marker::PhantomData;

// State tags — zero-sized types used only as type parameters
struct Disconnected;
struct Connected;
struct Authenticated;

// The connection type carries its state in a phantom type parameter
struct Connection<State> {
    host: String,
    port: u16,
    _state: PhantomData<State>,
}

// Methods only available in Disconnected state
impl Connection<Disconnected> {
    pub fn new(host: &str, port: u16) -> Self {
        Connection {
            host: host.to_string(),
            port,
            _state: PhantomData,
        }
    }

    // Consumes Disconnected, returns Connected (state transition)
    pub fn connect(self) -> Result<Connection<Connected>, String> {
        println!("connecting to {}:{}", self.host, self.port);
        // real TCP logic here
        Ok(Connection { host: self.host, port: self.port, _state: PhantomData })
    }
}

// Methods only available in Connected state
impl Connection<Connected> {
    pub fn authenticate(self, token: &str) -> Result<Connection<Authenticated>, String> {
        println!("authenticating with token: {token}");
        if token.is_empty() {
            Err("empty token".to_string())
        } else {
            Ok(Connection { host: self.host, port: self.port, _state: PhantomData })
        }
    }

    pub fn ping(&self) -> bool {
        println!("pinging {}:{}", self.host, self.port);
        true
    }
}

// Methods only available in Authenticated state
impl Connection<Authenticated> {
    pub fn send(&self, data: &[u8]) -> usize {
        println!("sending {} bytes", data.len());
        data.len()
    }

    pub fn subscribe(&self, topic: &str) {
        println!("subscribed to {topic}");
    }
}

fn main() {
    let conn = Connection::<Disconnected>::new("example.com", 443);
    // conn.send(b"hello"); // compile error: no method `send` on Connection<Disconnected>

    let conn = conn.connect().expect("connection failed");
    conn.ping(); // ok
    // conn.send(b"hello"); // compile error: no method `send` on Connection<Connected>

    let conn = conn.authenticate("secret-token").expect("auth failed");
    conn.send(b"hello world"); // ok
    conn.subscribe("events");
}

2.1 Typestate for Builder Validation

use std::marker::PhantomData;

struct Missing;
struct Present<T>(T);

struct RequestBuilder<Url, Method> {
    url: Url,
    method: Method,
    headers: Vec<(String, String)>,
    body: Option<Vec<u8>>,
}

impl RequestBuilder<Missing, Missing> {
    fn new() -> Self {
        RequestBuilder { url: Missing, method: Missing, headers: vec![], body: None }
    }
}

impl<M> RequestBuilder<Missing, M> {
    fn url(self, url: &str) -> RequestBuilder<Present<String>, M> {
        RequestBuilder {
            url: Present(url.to_string()),
            method: self.method,
            headers: self.headers,
            body: self.body,
        }
    }
}

impl<U> RequestBuilder<U, Missing> {
    fn get(self) -> RequestBuilder<U, Present<&'static str>> {
        RequestBuilder { url: self.url, method: Present("GET"), headers: self.headers, body: self.body }
    }
    fn post(self) -> RequestBuilder<U, Present<&'static str>> {
        RequestBuilder { url: self.url, method: Present("POST"), headers: self.headers, body: self.body }
    }
}

// build() only available when both Url and Method are Present
impl RequestBuilder<Present<String>, Present<&'static str>> {
    fn build(self) -> String {
        format!("{} {}", self.method.0, self.url.0)
    }
}

fn main() {
    let req = RequestBuilder::new()
        .url("https://api.example.com/users")
        .get()
        .build(); // compiles: url and method both present

    println!("{req}"); // GET https://api.example.com/users

    // RequestBuilder::new().build(); // compile error: missing url and method
    // RequestBuilder::new().url("x").build(); // compile error: missing method
}

3. RAII — Resource Acquisition Is Initialization

Rust's ownership system makes RAII automatic. Implement Drop to release resources when a value goes out of scope — no manual cleanup, no leaks.

use std::sync::{Arc, Mutex};

// A scoped timer that prints elapsed time when dropped
struct Timer {
    label: String,
    start: std::time::Instant,
}

impl Timer {
    fn new(label: &str) -> Self {
        println!("[{label}] started");
        Timer { label: label.to_string(), start: std::time::Instant::now() }
    }
}

impl Drop for Timer {
    fn drop(&mut self) {
        println!("[{}] elapsed: {:?}", self.label, self.start.elapsed());
    }
}

// A scoped lock that releases a resource when dropped
struct ScopedLock<'a, T> {
    guard: std::sync::MutexGuard<'a, T>,
    label: &'static str,
}

impl<'a, T> ScopedLock<'a, T> {
    fn acquire(mutex: &'a Mutex<T>, label: &'static str) -> Self {
        println!("[lock] acquiring {label}");
        ScopedLock { guard: mutex.lock().unwrap(), label }
    }
}

impl<T> Drop for ScopedLock<'_, T> {
    fn drop(&mut self) {
        println!("[lock] releasing {}", self.label);
    }
}

impl<T> std::ops::Deref for ScopedLock<'_, T> {
    type Target = T;
    fn deref(&self) -> &T { &self.guard }
}

fn main() {
    {
        let _t = Timer::new("operation");
        std::thread::sleep(std::time::Duration::from_millis(50));
        println!("doing work...");
        // _t dropped here — prints elapsed time
    }

    let data = Mutex::new(vec![1, 2, 3]);
    {
        let lock = ScopedLock::acquire(&data, "data");
        println!("data len: {}", lock.len()); // via Deref
        // lock dropped here — releases mutex
    }
    println!("mutex free again");
}

4. Extension Traits

Add methods to types you don't own — including standard library types — by defining a trait and implementing it. This sidesteps the orphan rule by defining the trait in your crate.

// Add methods to &[T] slices
pub trait SliceExt<T> {
    fn second(&self) -> Option<&T>;
    fn average(&self) -> f64 where T: Into<f64> + Copy;
    fn contains_any(&self, others: &[T]) -> bool where T: PartialEq;
}

impl<T> SliceExt<T> for [T] {
    fn second(&self) -> Option<&T> { self.get(1) }

    fn average(&self) -> f64 where T: Into<f64> + Copy {
        if self.is_empty() { return 0.0; }
        let sum: f64 = self.iter().map(|&x| x.into()).sum();
        sum / self.len() as f64
    }

    fn contains_any(&self, others: &[T]) -> bool where T: PartialEq {
        others.iter().any(|o| self.contains(o))
    }
}

// Add methods to String / &str
pub trait StrExt {
    fn truncate_at(&self, max: usize, suffix: &str) -> String;
    fn count_words(&self) -> usize;
}

impl StrExt for str {
    fn truncate_at(&self, max: usize, suffix: &str) -> String {
        if self.chars().count() <= max {
            self.to_string()
        } else {
            let cut: String = self.chars().take(max).collect();
            format!("{cut}{suffix}")
        }
    }

    fn count_words(&self) -> usize {
        self.split_whitespace().count()
    }
}

fn main() {
    let nums = [10u32, 20, 30, 40, 50];
    println!("second: {:?}", nums.second());       // Some(20)
    println!("average: {}", nums.average());        // 30
    println!("contains 30 or 99: {}", nums.contains_any(&[99, 30])); // true

    let s = "Hello, beautiful world!";
    println!("{}", s.truncate_at(10, "…")); // Hello, bea…
    println!("words: {}", s.count_words()); // 3
}

5. Sealed Traits

A sealed trait can only be implemented by types within your crate. Users can use the trait but not implement it — useful for extensible enums and stable APIs.

// The sealing mechanism: a private supertrait
mod private {
    pub trait Sealed {}
}

// Public trait with a private supertrait — external code cannot implement it
// because it cannot name `private::Sealed` to satisfy the bound
pub trait SqlType: private::Sealed {
    fn sql_type_name() -> &'static str;
    fn to_sql_string(&self) -> String;
}

// Only types in this module can impl SqlType
pub struct SqlInt(pub i64);
pub struct SqlText(pub String);
pub struct SqlBool(pub bool);

impl private::Sealed for SqlInt {}
impl SqlType for SqlInt {
    fn sql_type_name() -> &'static str { "INTEGER" }
    fn to_sql_string(&self) -> String { self.0.to_string() }
}

impl private::Sealed for SqlText {}
impl SqlType for SqlText {
    fn sql_type_name() -> &'static str { "TEXT" }
    fn to_sql_string(&self) -> String { format!("'{}'", self.0.replace('\'', "''")) }
}

impl private::Sealed for SqlBool {}
impl SqlType for SqlBool {
    fn sql_type_name() -> &'static str { "BOOLEAN" }
    fn to_sql_string(&self) -> String { (if self.0 { "TRUE" } else { "FALSE" }).to_string() }
}

fn bind<T: SqlType>(placeholder: &str, value: &T) -> String {
    format!("{placeholder} = {}", value.to_sql_string())
}

fn main() {
    println!("{}", bind("age",   &SqlInt(30)));           // age = 30
    println!("{}", bind("name",  &SqlText("O'Brian".to_string()))); // name = 'O''Brian'
    println!("{}", bind("admin", &SqlBool(true)));        // admin = TRUE
}

6. State Machine Pattern

Represent a state machine as an enum where each variant holds the data for that state. Transitions return a new variant.

#[derive(Debug)]
enum TrafficLight {
    Red { remaining_secs: u32 },
    Yellow { remaining_secs: u32 },
    Green { remaining_secs: u32 },
}

impl TrafficLight {
    fn new() -> Self {
        TrafficLight::Red { remaining_secs: 30 }
    }

    fn tick(self) -> Self {
        match self {
            TrafficLight::Red { remaining_secs: 1 } =>
                TrafficLight::Green { remaining_secs: 25 },
            TrafficLight::Red { remaining_secs } =>
                TrafficLight::Red { remaining_secs: remaining_secs - 1 },

            TrafficLight::Green { remaining_secs: 1 } =>
                TrafficLight::Yellow { remaining_secs: 5 },
            TrafficLight::Green { remaining_secs } =>
                TrafficLight::Green { remaining_secs: remaining_secs - 1 },

            TrafficLight::Yellow { remaining_secs: 1 } =>
                TrafficLight::Red { remaining_secs: 30 },
            TrafficLight::Yellow { remaining_secs } =>
                TrafficLight::Yellow { remaining_secs: remaining_secs - 1 },
        }
    }

    fn can_go(&self) -> bool {
        matches!(self, TrafficLight::Green { .. })
    }

    fn color(&self) -> &'static str {
        match self {
            TrafficLight::Red { .. }    => "🔴 Red",
            TrafficLight::Yellow { .. } => "🟡 Yellow",
            TrafficLight::Green { .. }  => "🟢 Green",
        }
    }
}

fn main() {
    let mut light = TrafficLight::new();
    for _ in 0..35 {
        println!("{} (can go: {})", light.color(), light.can_go());
        light = light.tick();
    }
}

7. Strategy Pattern — Trait Objects as Dependencies

// Define the strategy interface
trait Compressor: Send + Sync {
    fn compress(&self, data: &[u8]) -> Vec<u8>;
    fn name(&self) -> &'static str;
}

// Multiple strategies
struct NoOp;
struct Rle;

impl Compressor for NoOp {
    fn compress(&self, data: &[u8]) -> Vec<u8> { data.to_vec() }
    fn name(&self) -> &'static str { "none" }
}

impl Compressor for Rle {
    fn compress(&self, data: &[u8]) -> Vec<u8> {
        // Simple run-length encoding
        let mut out = Vec::new();
        let mut i = 0;
        while i < data.len() {
            let byte = data[i];
            let mut count = 1u8;
            while i + count as usize < data.len()
                && data[i + count as usize] == byte
                && count < 255
            {
                count += 1;
            }
            out.push(count);
            out.push(byte);
            i += count as usize;
        }
        out
    }
    fn name(&self) -> &'static str { "rle" }
}

// Consumer uses Box<dyn Compressor> — strategy chosen at runtime
struct Archiver {
    compressor: Box<dyn Compressor>,
}

impl Archiver {
    fn new(compressor: impl Compressor + 'static) -> Self {
        Archiver { compressor: Box::new(compressor) }
    }

    fn archive(&self, data: &[u8]) -> Vec<u8> {
        let compressed = self.compressor.compress(data);
        println!("  algorithm: {}, {}->{} bytes", self.compressor.name(), data.len(), compressed.len());
        compressed
    }
}

// Or use generics for zero-cost monomorphization (strategy known at compile time)
struct StaticArchiver<C: Compressor> {
    compressor: C,
}

fn main() {
    let data = b"aaaaaabbbcccccdddddddddd";

    let a1 = Archiver::new(NoOp);
    let a2 = Archiver::new(Rle);

    a1.archive(data); // algorithm: none, 24->24 bytes
    a2.archive(data); // algorithm: rle, 24->8 bytes
}

8. Visitor Pattern

Separate algorithms from the data structures they operate on. In Rust, this is often done with trait objects or enums + match.

// AST representation
#[derive(Debug)]
enum Expr {
    Number(f64),
    Variable(String),
    BinOp { op: char, left: Box<Expr>, right: Box<Expr> },
    Negate(Box<Expr>),
}

// Visitor trait — one method per node type
trait ExprVisitor {
    type Output;
    fn visit_number(&mut self, n: f64) -> Self::Output;
    fn visit_variable(&mut self, name: &str) -> Self::Output;
    fn visit_binop(&mut self, op: char, left: &Expr, right: &Expr) -> Self::Output;
    fn visit_negate(&mut self, inner: &Expr) -> Self::Output;
}

impl Expr {
    fn accept<V: ExprVisitor>(&self, v: &mut V) -> V::Output {
        match self {
            Expr::Number(n)              => v.visit_number(*n),
            Expr::Variable(name)         => v.visit_variable(name),
            Expr::BinOp { op, left, right } => v.visit_binop(*op, left, right),
            Expr::Negate(inner)          => v.visit_negate(inner),
        }
    }
}

// Evaluator visitor
struct Evaluator {
    vars: std::collections::HashMap<String, f64>,
}

impl ExprVisitor for Evaluator {
    type Output = f64;

    fn visit_number(&mut self, n: f64) -> f64 { n }

    fn visit_variable(&mut self, name: &str) -> f64 {
        *self.vars.get(name).expect("undefined variable")
    }

    fn visit_binop(&mut self, op: char, left: &Expr, right: &Expr) -> f64 {
        let l = left.accept(self);
        let r = right.accept(self);
        match op {
            '+' => l + r, '-' => l - r,
            '*' => l * r, '/' => l / r,
            _   => panic!("unknown op"),
        }
    }

    fn visit_negate(&mut self, inner: &Expr) -> f64 { -inner.accept(self) }
}

// Pretty-printer visitor
struct Printer;

impl ExprVisitor for Printer {
    type Output = String;
    fn visit_number(&mut self, n: f64) -> String { n.to_string() }
    fn visit_variable(&mut self, name: &str) -> String { name.to_string() }
    fn visit_binop(&mut self, op: char, l: &Expr, r: &Expr) -> String {
        format!("({} {op} {})", l.accept(self), r.accept(self))
    }
    fn visit_negate(&mut self, inner: &Expr) -> String {
        format!("(-{})", inner.accept(self))
    }
}

fn main() {
    // (x + 2) * -(y - 1)
    let ast = Expr::BinOp {
        op: '*',
        left: Box::new(Expr::BinOp {
            op: '+',
            left: Box::new(Expr::Variable("x".to_string())),
            right: Box::new(Expr::Number(2.0)),
        }),
        right: Box::new(Expr::Negate(Box::new(Expr::BinOp {
            op: '-',
            left: Box::new(Expr::Variable("y".to_string())),
            right: Box::new(Expr::Number(1.0)),
        }))),
    };

    let printed = ast.accept(&mut Printer);
    println!("{printed}"); // ((x + 2) * (-(y - 1)))

    let mut eval = Evaluator {
        vars: [("x".to_string(), 3.0), ("y".to_string(), 5.0)].into(),
    };
    let result = ast.accept(&mut eval);
    println!("= {result}"); // = -20
}

9. Command Pattern — Undo/Redo

trait Command {
    fn execute(&mut self, text: &mut String);
    fn undo(&mut self, text: &mut String);
    fn description(&self) -> String;
}

struct Insert { pos: usize, content: String }
struct Delete { pos: usize, len: usize, deleted: String }

impl Command for Insert {
    fn execute(&mut self, text: &mut String) {
        text.insert_str(self.pos, &self.content);
    }
    fn undo(&mut self, text: &mut String) {
        text.drain(self.pos..self.pos + self.content.len());
    }
    fn description(&self) -> String {
        format!("insert {:?} at {}", self.content, self.pos)
    }
}

impl Command for Delete {
    fn execute(&mut self, text: &mut String) {
        self.deleted = text.drain(self.pos..self.pos + self.len).collect();
    }
    fn undo(&mut self, text: &mut String) {
        text.insert_str(self.pos, &self.deleted);
    }
    fn description(&self) -> String {
        format!("delete {} chars at {}", self.len, self.pos)
    }
}

struct Editor {
    text: String,
    history: Vec<Box<dyn Command>>,
}

impl Editor {
    fn new() -> Self { Editor { text: String::new(), history: Vec::new() } }

    fn execute(&mut self, mut cmd: Box<dyn Command>) {
        cmd.execute(&mut self.text);
        println!("  did: {}", cmd.description());
        self.history.push(cmd);
    }

    fn undo(&mut self) {
        if let Some(mut cmd) = self.history.pop() {
            cmd.undo(&mut self.text);
            println!("  undid: {}", cmd.description());
        }
    }
}

fn main() {
    let mut ed = Editor::new();
    ed.execute(Box::new(Insert { pos: 0, content: "Hello".to_string() }));
    ed.execute(Box::new(Insert { pos: 5, content: ", world!".to_string() }));
    println!("text: {:?}", ed.text); // "Hello, world!"

    ed.execute(Box::new(Delete { pos: 5, len: 8, deleted: String::new() }));
    println!("text: {:?}", ed.text); // "Hello"

    ed.undo(); // restore ", world!"
    println!("text: {:?}", ed.text); // "Hello, world!"

    ed.undo();
    println!("text: {:?}", ed.text); // "Hello"
}

10. Summary: When to Use Each Pattern

Pattern Use when
NewtypePrevent mixing semantically different values of the same underlying type
TypestateEnforce API call ordering at compile time
RAII + DropAutomatic cleanup of any resource (files, locks, timers, DB connections)
Extension traitAdd methods to foreign types without orphan rule violations
Sealed traitExpose a trait publicly but restrict implementations to your crate
State machineModel objects with well-defined transitions (protocols, workflows, UIs)
StrategySwap algorithms at runtime (Box<dyn Trait>) or compile time (generics)
VisitorApply multiple algorithms to an AST or recursive data structure
CommandImplement undo/redo, queuing, or logging of operations
