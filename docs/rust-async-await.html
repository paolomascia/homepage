Async/Await & Futures in Rust

Rust Async Futures

1. The Future Trait

A Future represents a value that may not be available yet. It is a state machine that the executor drives by calling poll() repeatedly.

use std::task::{Context, Poll};
use std::pin::Pin;

// The core trait from std::future
pub trait Future {
    type Output;

    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output>;
}

// Poll is a simple enum:
// enum Poll<T> {
//     Ready(T),   — computation complete, here's the value
//     Pending,    — not ready yet; will call cx.waker().wake() when ready
// }

Key facts about futures in Rust:

  • Futures are lazy — nothing happens until an executor calls poll().
  • When a future returns Pending, it must arrange to call cx.waker().wake() when it may be ready, so the executor knows to poll again.
  • The .await keyword desugars into repeated poll() calls.

1.1 Implementing a Simple Future Manually

use std::future::Future;
use std::pin::Pin;
use std::task::{Context, Poll};
use std::time::{Duration, Instant};

// A future that resolves after a certain duration (busy-waits for illustration)
struct Delay {
    deadline: Instant,
}

impl Delay {
    fn new(duration: Duration) -> Self {
        Delay { deadline: Instant::now() + duration }
    }
}

impl Future for Delay {
    type Output = ();

    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<()> {
        if Instant::now() >= self.deadline {
            Poll::Ready(())
        } else {
            // In real code: register with the I/O driver and call wake() later.
            // Here we busy-wake (not production-quality).
            cx.waker().wake_by_ref();
            Poll::Pending
        }
    }
}

// Usage: requires an async runtime to drive the future
// #[tokio::main]
// async fn main() {
//     Delay::new(Duration::from_millis(100)).await;
//     println!("done");
// }

2. async fn and .await

An async fn is sugar for a function that returns impl Future<Output = T>. The compiler transforms the body into a state machine automatically.

// Cargo.toml: tokio = { version = "1", features = ["full"] }
use tokio::time::{sleep, Duration};

// `async fn` returns an impl Future — calling it does NOT start execution
async fn fetch_data(id: u32) -> String {
    sleep(Duration::from_millis(10)).await; // suspend here, yield to executor
    format!("data-{id}")
}

// The #[tokio::main] macro starts the Tokio runtime and drives the top-level future
#[tokio::main]
async fn main() {
    // .await drives the future to completion, suspending the async task if needed
    let result = fetch_data(42).await;
    println!("{result}"); // data-42
}

2.1 Desugared View

// This async fn:
async fn add(a: i32, b: i32) -> i32 {
    a + b
}

// Is roughly equivalent to:
fn add_desugared(a: i32, b: i32) -> impl std::future::Future<Output = i32> {
    async move { a + b }
}

// The compiler generates an anonymous state-machine struct that implements Future.
// Each .await point becomes a state transition in that machine.

2.2 Sequential vs Concurrent .await

use tokio::time::{sleep, Duration, Instant};

async fn work(label: &str, ms: u64) -> String {
    sleep(Duration::from_millis(ms)).await;
    label.to_string()
}

#[tokio::main]
async fn main() {
    // Sequential: total time = 100 + 200 + 300 = 600ms
    let start = Instant::now();
    let a = work("a", 100).await;
    let b = work("b", 200).await;
    let c = work("c", 300).await;
    println!("sequential: {}ms — {a}, {b}, {c}", start.elapsed().as_millis());

    // Concurrent: total time = max(100, 200, 300) = 300ms
    let start = Instant::now();
    let (a, b, c) = tokio::join!(work("a", 100), work("b", 200), work("c", 300));
    println!("concurrent: {}ms — {a}, {b}, {c}", start.elapsed().as_millis());
}
// sequential: ~600ms — a, b, c
// concurrent: ~300ms — a, b, c

3. Concurrency Combinators

3.1 join! — Wait for All

use tokio::time::{sleep, Duration};

async fn fetch(url: &str) -> String {
    sleep(Duration::from_millis(50)).await;
    format!("response from {url}")
}

#[tokio::main]
async fn main() {
    // join! polls all futures concurrently on the same task
    let (r1, r2, r3) = tokio::join!(
        fetch("api.example.com/users"),
        fetch("api.example.com/posts"),
        fetch("api.example.com/comments"),
    );
    println!("{r1}");
    println!("{r2}");
    println!("{r3}");
}

3.2 try_join! — Short-circuit on Error

use tokio::time::{sleep, Duration};

async fn fetch_json(id: u32) -> Result<String, String> {
    sleep(Duration::from_millis(20)).await;
    if id == 0 {
        Err(format!("invalid id: {id}"))
    } else {
        Ok(format!("{{\"id\": {id}}}}"))
    }
}

#[tokio::main]
async fn main() {
    // try_join! returns Err immediately if any future fails
    match tokio::try_join!(fetch_json(1), fetch_json(0), fetch_json(3)) {
        Ok((a, b, c)) => println!("ok: {a}, {b}, {c}"),
        Err(e)         => println!("error: {e}"), // "invalid id: 0"
    }

    // All success
    match tokio::try_join!(fetch_json(1), fetch_json(2)) {
        Ok((a, b)) => println!("ok: {a}, {b}"),
        Err(e)     => println!("error: {e}"),
    }
}

3.3 select! — First One Wins

use tokio::time::{sleep, Duration};

async fn slow_task() -> &'static str { sleep(Duration::from_millis(500)).await; "slow" }
async fn fast_task() -> &'static str { sleep(Duration::from_millis(50)).await;  "fast" }

#[tokio::main]
async fn main() {
    // select! takes the first future to complete and cancels the rest
    let winner = tokio::select! {
        v = slow_task() => v,
        v = fast_task() => v,
    };
    println!("winner: {winner}"); // "fast"

    // Common pattern: cancellation via select! with a shutdown signal
    let (shutdown_tx, mut shutdown_rx) = tokio::sync::oneshot::channel::<()>();

    tokio::spawn(async move {
        sleep(Duration::from_millis(100)).await;
        let _ = shutdown_tx.send(());
    });

    tokio::select! {
        _ = &mut shutdown_rx => println!("shutdown received"),
        _ = slow_task() => println!("task completed"),
    }
}

4. Spawning Tasks

tokio::spawn creates an independent async task (green thread). Unlike .await, spawned tasks run concurrently with the current task without blocking it.

use tokio::time::{sleep, Duration};

#[tokio::main]
async fn main() {
    // spawn returns a JoinHandle — you can await it to get the result
    let handle = tokio::spawn(async {
        sleep(Duration::from_millis(50)).await;
        println!("spawned task done");
        42u32
    });

    println!("main task continues immediately");

    // await the handle to get the result
    let value = handle.await.unwrap(); // unwrap the Result (Err if the task panicked)
    println!("spawned returned: {value}");

    // Spawning many tasks concurrently
    let handles: Vec<_> = (0..10)
        .map(|i| {
            tokio::spawn(async move {
                sleep(Duration::from_millis(i * 10)).await;
                i * i
            })
        })
        .collect();

    let results: Vec<u64> = futures::future::join_all(handles)
        .await
        .into_iter()
        .map(|r| r.unwrap())
        .collect();

    println!("{results:?}"); // [0, 1, 4, 9, 16, 25, 36, 49, 64, 81]
}

Note: spawned tasks must be 'static and Send (because Tokio may move them between threads in a multi-threaded runtime).

5. Pin and Unpin

Async state machines can contain self-referential pointers (e.g., a reference into a local buffer). Moving such a struct would invalidate the pointer. Pin<P> prevents the value from being moved after it has been pinned.

use std::pin::Pin;
use std::future::Future;

// Most types implement Unpin — pinning them is a no-op
// Types with self-references do NOT implement Unpin by default

// Stack-pinning with pin! macro (tokio re-exports it)
async fn stack_pin_example() {
    let mut future = std::pin::pin!(async {
        // This future may be self-referential internally
        let x = 42;
        std::future::ready(()).await;
        x // x referenced after an await point
    });

    // future is now pinned on the stack and can be polled
    let result = future.await;
    println!("{result}");
}

// Heap-pinning with Box::pin
fn heap_pin_example() -> Pin<Box<dyn Future<Output = i32> + Send>> {
    Box::pin(async { 100 })
}

// When you need to store a future in a struct field
struct TaskRunner {
    // Without Pin: cannot store a &mut reference to the future safely
    // With Pin: the future cannot be moved after polling starts
    future: Pin<Box<dyn Future<Output = ()> + Send>>,
}

impl TaskRunner {
    fn new(f: impl Future<Output = ()> + Send + 'static) -> Self {
        TaskRunner { future: Box::pin(f) }
    }
}

5.1 Practical Rules

  • If you just .await futures — you never need to think about Pin.
  • If you store futures in structs or implement Future manually — use Box::pin().
  • If you use tokio::select! with futures by reference — use tokio::pin!() or std::pin::pin!().

6. Async Traits

As of Rust 1.75, async fn in traits is stable. Earlier code used the async-trait crate.

// Stable async fn in traits (Rust 1.75+)
trait DataStore {
    async fn fetch(&self, key: &str) -> Option<String>;
    async fn store(&mut self, key: &str, value: String);
}

struct MemStore(std::collections::HashMap<String, String>);

impl DataStore for MemStore {
    async fn fetch(&self, key: &str) -> Option<String> {
        // async fn in impl can use .await
        self.0.get(key).cloned()
    }

    async fn store(&mut self, key: &str, value: String) {
        self.0.insert(key.to_string(), value);
    }
}

// For trait objects (dyn DataStore), use async-trait crate or return-position
// impl Trait (RPIT) workarounds — async fn in trait objects is still limited.

6.1 async-trait for Older Code / dyn Dispatch

// Cargo.toml: async-trait = "0.1"
use async_trait::async_trait;

#[async_trait]
trait Fetcher: Send + Sync {
    async fn fetch(&self, url: &str) -> Result<String, String>;
}

struct HttpFetcher;

#[async_trait]
impl Fetcher for HttpFetcher {
    async fn fetch(&self, url: &str) -> Result<String, String> {
        // In real code: reqwest::get(url).await...
        Ok(format!("response from {url}"))
    }
}

// Now usable as a trait object — async-trait boxes the returned future
async fn run(fetcher: &dyn Fetcher) {
    let resp = fetcher.fetch("https://example.com").await.unwrap();
    println!("{resp}");
}

7. Async Streams

An async stream is like an iterator that yields values asynchronously. The futures crate provides StreamExt with familiar combinators.

// Cargo.toml: futures = "0.3", tokio = { version = "1", features = ["full"] }
use futures::stream::{self, StreamExt};
use tokio::time::{sleep, Duration};

#[tokio::main]
async fn main() {
    // Create a stream from an iterator
    let stream = stream::iter(0u32..10);

    // StreamExt provides async versions of iterator combinators
    let result: Vec<u32> = stream
        .filter(|&n| async move { n % 2 == 0 })
        .map(|n| async move { n * n })
        .buffered(4)          // drive up to 4 futures concurrently
        .collect()
        .await;

    println!("{result:?}"); // [0, 4, 16, 36, 64]

    // Async generator — emit values over time
    let ticks = stream::unfold(0u32, |i| async move {
        if i >= 5 { return None; }
        sleep(Duration::from_millis(20)).await;
        Some((i, i + 1))
    });

    tokio::pin!(ticks);
    while let Some(v) = ticks.next().await {
        print!("{v} ");
    }
    println!();
}

7.1 tokio_stream and ReceiverStream

// Cargo.toml: tokio-stream = "0.1"
use tokio::sync::mpsc;
use tokio_stream::{wrappers::ReceiverStream, StreamExt};

#[tokio::main]
async fn main() {
    let (tx, rx) = mpsc::channel::<i32>(16);

    tokio::spawn(async move {
        for i in 0..5 {
            tx.send(i).await.unwrap();
        }
    });

    // Wrap a tokio mpsc receiver into a Stream
    let mut stream = ReceiverStream::new(rx);

    while let Some(item) = stream.next().await {
        println!("received: {item}");
    }
}

8. Error Handling in Async Code

use tokio::time::{sleep, Duration};
use std::io;

// async functions return Result like regular functions
async fn read_file(path: &str) -> Result<String, io::Error> {
    // tokio::fs is the async equivalent of std::fs
    tokio::fs::read_to_string(path).await
}

async fn fetch_with_retry(url: &str, retries: u32) -> Result<String, String> {
    let mut last_err = String::new();
    for attempt in 0..=retries {
        if attempt > 0 {
            sleep(Duration::from_millis(100 * 2u64.pow(attempt))).await; // backoff
        }
        match simulate_fetch(url).await {
            Ok(resp) => return Ok(resp),
            Err(e)   => last_err = e,
        }
    }
    Err(format!("all retries failed: {last_err}"))
}

async fn simulate_fetch(url: &str) -> Result<String, String> {
    sleep(Duration::from_millis(10)).await;
    if url.contains("bad") {
        Err("connection refused".to_string())
    } else {
        Ok(format!("200 OK from {url}"))
    }
}

#[tokio::main]
async fn main() {
    match fetch_with_retry("https://good.example.com", 3).await {
        Ok(r)  => println!("{r}"),
        Err(e) => eprintln!("error: {e}"),
    }

    match fetch_with_retry("https://bad.example.com", 2).await {
        Ok(r)  => println!("{r}"),
        Err(e) => eprintln!("error: {e}"),
    }
}

9. Common Async Pitfalls

9.1 Blocking Inside async — Don't!

use tokio::time::{sleep, Duration};
use std::thread;

// BAD: std::thread::sleep blocks the entire executor thread
async fn bad_delay(ms: u64) {
    thread::sleep(Duration::from_millis(ms)); // blocks all async tasks on this thread!
}

// GOOD: use the async equivalent
async fn good_delay(ms: u64) {
    sleep(Duration::from_millis(ms)).await; // yields to executor
}

// For CPU-bound work, use spawn_blocking to offload to a thread pool
async fn cpu_intensive(data: Vec<u32>) -> Vec<u32> {
    tokio::task::spawn_blocking(move || {
        // runs on a dedicated blocking thread, does not block the async executor
        let mut d = data;
        d.sort();
        d
    })
    .await
    .unwrap()
}

9.2 Holding a Mutex Guard Across .await

use std::sync::Mutex;
use tokio::time::{sleep, Duration};

// BAD: std::sync::MutexGuard held across .await — may cause deadlock
// and won't even compile with Tokio's multi-thread runtime
async fn bad_lock(m: &Mutex<i32>) {
    let _guard = m.lock().unwrap(); // std::sync::MutexGuard is not Send
    // sleep(Duration::from_millis(10)).await; // compile error in multi-thread mode
}

// GOOD 1: drop the guard before .await
async fn good_lock_v1(m: &Mutex<i32>) -> i32 {
    let value = {
        let guard = m.lock().unwrap();
        *guard // copy the value, guard dropped at end of block
    };
    sleep(Duration::from_millis(10)).await; // no lock held
    value
}

// GOOD 2: use tokio::sync::Mutex which IS Send
async fn good_lock_v2(m: &tokio::sync::Mutex<i32>) -> i32 {
    let mut guard = m.lock().await; // async lock — yields instead of blocking
    *guard += 1;
    sleep(Duration::from_millis(10)).await; // OK: tokio::MutexGuard is Send
    *guard
}

9.3 Async is Concurrent, Not Parallel by Default

use tokio::time::{sleep, Duration, Instant};

#[tokio::main(flavor = "current_thread")] // single-threaded runtime
async fn main() {
    let start = Instant::now();

    // join! runs futures concurrently (interleaved) on ONE thread
    // For true parallelism use tokio::spawn (multi-thread runtime)
    let (a, b) = tokio::join!(
        async { sleep(Duration::from_millis(100)).await; "a" },
        async { sleep(Duration::from_millis(100)).await; "b" },
    );

    // ~100ms total (concurrent), not 200ms (sequential), even on 1 thread
    println!("done in {}ms: {a}, {b}", start.elapsed().as_millis());
}

10. Executors — The Runtime

A Rust future does nothing on its own. An executor drives it by polling. The most popular executor is Tokio; alternatives include async-std, smol, and embassy (for embedded).

// Minimal executor from scratch (educational, not production)
use std::future::Future;
use std::task::{Context, Poll, RawWaker, RawWakerVTable, Waker};
use std::pin::pin;

fn noop_waker() -> Waker {
    // A waker that does nothing (fine for busy-polling)
    const VTABLE: RawWakerVTable = RawWakerVTable::new(
        |_| RawWaker::new(std::ptr::null(), &VTABLE), // clone
        |_| {},  // wake
        |_| {},  // wake_by_ref
        |_| {},  // drop
    );
    unsafe { Waker::from_raw(RawWaker::new(std::ptr::null(), &VTABLE)) }
}

fn block_on<F: Future>(mut future: F) -> F::Output {
    let waker = noop_waker();
    let mut cx = Context::from_waker(&waker);
    let mut future = pin!(future);

    loop {
        match future.as_mut().poll(&mut cx) {
            Poll::Ready(v) => return v,
            Poll::Pending  => { /* spin — real executors sleep here */ }
        }
    }
}

fn main() {
    let result = block_on(async {
        // no .await across OS yields needed here for this simple executor
        1 + 1
    });
    println!("result: {result}"); // 2
}

In practice, always use #[tokio::main] or tokio::runtime::Runtime::new() — never write your own executor for production code.
