Reqwest HTTP Client

Rust Reqwest HTTP

1. Setup and Basic Requests

# Cargo.toml
[dependencies]
reqwest = { version = "0.12", features = ["json", "stream", "multipart"] }
tokio   = { version = "1", features = ["full"] }
serde   = { version = "1", features = ["derive"] }
anyhow  = "1"

use reqwest::Client;
use serde::{Deserialize, Serialize};
use anyhow::Result;

#[derive(Debug, Deserialize)]
struct Post {
    id: u32,
    title: String,
    body: String,
    #[serde(rename = "userId")]
    user_id: u32,
}

#[tokio::main]
async fn main() -> Result<()> {
    // Create a client once — reuse across requests (connection pool!)
    let client = Client::new();

    // Simple GET — deserialize JSON directly
    let post: Post = client
        .get("https://jsonplaceholder.typicode.com/posts/1")
        .send()
        .await?
        .error_for_status()?   // returns Err for 4xx/5xx
        .json()
        .await?;

    println!("{post:#?}");

    // GET with query parameters
    let posts: Vec<Post> = client
        .get("https://jsonplaceholder.typicode.com/posts")
        .query(&[("userId", "1"), ("_limit", "3")])
        .send()
        .await?
        .json()
        .await?;

    println!("found {} posts", posts.len());

    Ok(())
}

2. POST, PUT, PATCH, DELETE

use reqwest::Client;
use serde::{Deserialize, Serialize};
use anyhow::Result;

#[derive(Serialize)]
struct CreatePost {
    title: String,
    body: String,
    user_id: u32,
}

#[derive(Debug, Deserialize)]
struct CreatedPost {
    id: u32,
    title: String,
}

async fn crud_example(client: &Client) -> Result<()> {
    let base = "https://jsonplaceholder.typicode.com";

    // POST with JSON body
    let new_post = CreatePost {
        title: "My Post".to_string(),
        body: "Hello, world!".to_string(),
        user_id: 1,
    };

    let created: CreatedPost = client
        .post(format!("{base}/posts"))
        .json(&new_post)          // sets Content-Type: application/json
        .send()
        .await?
        .error_for_status()?
        .json()
        .await?;

    println!("created: {created:?}");

    // PUT — full replacement
    let resp = client
        .put(format!("{base}/posts/1"))
        .json(&serde_json::json!({ "title": "Updated", "body": "New body", "userId": 1 }))
        .send()
        .await?
        .error_for_status()?;

    println!("put: {}", resp.status());

    // PATCH — partial update
    let resp = client
        .patch(format!("{base}/posts/1"))
        .json(&serde_json::json!({ "title": "Patched title" }))
        .send()
        .await?;

    println!("patch: {}", resp.status());

    // DELETE
    let resp = client
        .delete(format!("{base}/posts/1"))
        .send()
        .await?
        .error_for_status()?;

    println!("deleted: {}", resp.status()); // 200

    Ok(())
}

3. Headers and Authentication

use reqwest::{Client, header};
use anyhow::Result;

async fn auth_examples() -> Result<()> {
    // Bearer token
    let client = Client::new();

    let resp = client
        .get("https://api.example.com/protected")
        .bearer_auth("my-jwt-token")
        .send()
        .await?;

    // Basic auth
    let resp = client
        .get("https://api.example.com/resource")
        .basic_auth("username", Some("password"))
        .send()
        .await?;

    // Custom headers
    let resp = client
        .get("https://api.example.com/data")
        .header("x-api-key", "my-api-key")
        .header("x-request-id", "abc-123")
        .header(header::ACCEPT, "application/json")
        .send()
        .await?;

    // Client with default headers (applied to every request)
    let mut headers = header::HeaderMap::new();
    headers.insert("x-api-key", "my-key".parse().unwrap());
    headers.insert(header::USER_AGENT, "my-client/1.0".parse().unwrap());

    let client = Client::builder()
        .default_headers(headers)
        .build()?;

    let resp = client.get("https://api.example.com/anything").send().await?;
    println!("status: {}", resp.status());

    Ok(())
}

4. Reusable Client with Configuration

Build the client once with a timeout, TLS, and proxy settings; reuse it throughout the application lifetime.

use reqwest::{Client, header, redirect::Policy};
use std::time::Duration;
use anyhow::Result;

fn build_client(api_key: &str, base_url: &str) -> Result<Client> {
    let mut headers = header::HeaderMap::new();
    headers.insert("x-api-key", api_key.parse()?);
    headers.insert(header::ACCEPT, "application/json".parse()?);
    headers.insert(header::CONTENT_TYPE, "application/json".parse()?);

    let client = Client::builder()
        .default_headers(headers)
        .timeout(Duration::from_secs(30))           // request-level timeout
        .connect_timeout(Duration::from_secs(10))   // TCP connection timeout
        .pool_max_idle_per_host(10)                 // connection pool size
        .redirect(Policy::limited(5))               // follow up to 5 redirects
        .gzip(true)                                 // decompress gzip responses
        .user_agent("my-app/1.0")
        .build()?;

    Ok(client)
}

// Wrap the client for a specific API
struct ApiClient {
    client: Client,
    base_url: String,
}

impl ApiClient {
    fn new(api_key: &str, base_url: &str) -> Result<Self> {
        Ok(ApiClient {
            client: build_client(api_key, base_url)?,
            base_url: base_url.to_string(),
        })
    }

    async fn get_user(&self, id: u64) -> Result<serde_json::Value> {
        Ok(self.client
            .get(format!("{}/users/{id}", self.base_url))
            .send()
            .await?
            .error_for_status()?
            .json()
            .await?)
    }

    async fn create_user(&self, name: &str, email: &str) -> Result<serde_json::Value> {
        Ok(self.client
            .post(format!("{}/users", self.base_url))
            .json(&serde_json::json!({ "name": name, "email": email }))
            .send()
            .await?
            .error_for_status()?
            .json()
            .await?)
    }
}

5. Retry with Exponential Backoff

use reqwest::{Client, StatusCode};
use std::time::Duration;
use anyhow::{Result, bail};

async fn fetch_with_retry(
    client: &Client,
    url: &str,
    max_attempts: u32,
) -> Result<serde_json::Value> {
    let mut last_err = anyhow::anyhow!("no attempts made");

    for attempt in 0..max_attempts {
        if attempt > 0 {
            let delay = Duration::from_millis(100 * 2u64.pow(attempt - 1)); // 100ms, 200ms, 400ms…
            tokio::time::sleep(delay).await;
            eprintln!("retry {attempt}/{max_attempts} after {:?}", delay);
        }

        match client.get(url).send().await {
            Ok(resp) => {
                let status = resp.status();

                // Don't retry client errors (4xx) — they won't change
                if status.is_client_error() {
                    bail!("client error {status}: {}", resp.text().await.unwrap_or_default());
                }

                // Retry server errors (5xx) and rate limits (429)
                if status.is_server_error() || status == StatusCode::TOO_MANY_REQUESTS {
                    // Respect Retry-After header if present
                    if let Some(retry_after) = resp.headers().get("retry-after") {
                        if let Ok(secs) = retry_after.to_str().unwrap_or("0").parse::<u64>() {
                            tokio::time::sleep(Duration::from_secs(secs)).await;
                        }
                    }
                    last_err = anyhow::anyhow!("server error {status}");
                    continue;
                }

                return Ok(resp.json().await?);
            }
            Err(e) if e.is_connect() || e.is_timeout() => {
                last_err = e.into();
            }
            Err(e) => return Err(e.into()),
        }
    }

    Err(last_err.context(format!("failed after {max_attempts} attempts")))
}

6. Streaming Responses

use reqwest::Client;
use tokio::io::AsyncWriteExt;
use futures::StreamExt;
use anyhow::Result;

// Download a large file with progress
async fn download_file(client: &Client, url: &str, dest: &str) -> Result<()> {
    let resp = client.get(url).send().await?.error_for_status()?;

    let total = resp.content_length();
    let mut stream = resp.bytes_stream();
    let mut file = tokio::fs::File::create(dest).await?;
    let mut downloaded = 0u64;

    while let Some(chunk) = stream.next().await {
        let chunk = chunk?;
        file.write_all(&chunk).await?;
        downloaded += chunk.len() as u64;

        if let Some(total) = total {
            print!("\r{:.1}%", downloaded as f64 / total as f64 * 100.0);
        }
    }
    file.flush().await?;
    println!("\ndownloaded {} bytes to {dest}", downloaded);

    Ok(())
}

// Server-Sent Events (SSE) / streaming text
async fn stream_sse(client: &Client, url: &str) -> Result<()> {
    let resp = client
        .get(url)
        .header("accept", "text/event-stream")
        .send()
        .await?;

    let mut stream = resp.bytes_stream();
    while let Some(chunk) = stream.next().await {
        let chunk = chunk?;
        let text = std::str::from_utf8(&chunk)?;
        for line in text.lines() {
            if let Some(data) = line.strip_prefix("data: ") {
                println!("event: {data}");
            }
        }
    }

    Ok(())
}

7. Multipart Form Upload

use reqwest::{Client, multipart};
use anyhow::Result;

async fn upload_image(client: &Client, path: &str) -> Result<serde_json::Value> {
    let file = tokio::fs::read(path).await?;
    let file_name = std::path::Path::new(path)
        .file_name().unwrap().to_string_lossy().to_string();

    let part = multipart::Part::bytes(file)
        .file_name(file_name)
        .mime_str("image/jpeg")?;

    let form = multipart::Form::new()
        .text("title", "my image")
        .text("description", "uploaded via reqwest")
        .part("file", part);

    let resp = client
        .post("https://api.example.com/upload")
        .multipart(form)
        .send()
        .await?
        .error_for_status()?
        .json()
        .await?;

    Ok(resp)
}

8. Mocking for Tests

# Cargo.toml (dev-dependencies)
[dev-dependencies]
mockito = "1"
wiremock = "0.6"

// Using wiremock for integration tests
#[cfg(test)]
mod tests {
    use wiremock::{MockServer, Mock, ResponseTemplate};
    use wiremock::matchers::{method, path, header};
    use reqwest::Client;

    #[tokio::test]
    async fn test_get_user() {
        // Start a mock HTTP server bound to a random port
        let server = MockServer::start().await;

        Mock::given(method("GET"))
            .and(path("/users/1"))
            .and(header("x-api-key", "test-key"))
            .respond_with(
                ResponseTemplate::new(200)
                    .set_body_json(serde_json::json!({
                        "id": 1,
                        "name": "Alice",
                        "email": "alice@example.com"
                    }))
            )
            .expect(1) // assert it's called exactly once
            .mount(&server)
            .await;

        let client = reqwest::Client::builder()
            .build()
            .unwrap();

        let resp = client
            .get(format!("{}/users/1", server.uri()))
            .header("x-api-key", "test-key")
            .send()
            .await
            .unwrap();

        assert_eq!(resp.status(), 200);
        let body: serde_json::Value = resp.json().await.unwrap();
        assert_eq!(body["name"], "Alice");

        // wiremock verifies that .expect(1) was satisfied when MockServer drops
    }

    #[tokio::test]
    async fn test_retry_on_server_error() {
        let server = MockServer::start().await;

        // First call returns 503, second returns 200
        Mock::given(method("GET")).and(path("/data"))
            .respond_with(ResponseTemplate::new(503))
            .up_to_n_times(1)
            .mount(&server)
            .await;

        Mock::given(method("GET")).and(path("/data"))
            .respond_with(
                ResponseTemplate::new(200)
                    .set_body_json(serde_json::json!({ "ok": true }))
            )
            .mount(&server)
            .await;

        let client = reqwest::Client::new();
        let result = super::fetch_with_retry(&client, &format!("{}/data", server.uri()), 3).await;
        assert!(result.is_ok());
    }
}

9. Concurrent Requests

use reqwest::Client;
use futures::future;
use anyhow::Result;

async fn fetch_many(client: &Client, ids: &[u32]) -> Result<Vec<serde_json::Value>> {
    let futures: Vec<_> = ids.iter().map(|&id| {
        let client = client.clone();
        async move {
            client
                .get(format!("https://jsonplaceholder.typicode.com/posts/{id}"))
                .send()
                .await?
                .json::<serde_json::Value>()
                .await
        }
    }).collect();

    // Wait for all futures concurrently (not sequentially)
    let results = future::join_all(futures).await;
    results.into_iter().collect()  // Vec<Result> -> Result<Vec>
}

// Limit concurrency with a semaphore
async fn fetch_bounded(client: &Client, ids: &[u32], concurrency: usize) -> Vec<serde_json::Value> {
    use tokio::sync::Semaphore;
    use std::sync::Arc;

    let sem = Arc::new(Semaphore::new(concurrency));
    let mut handles = Vec::new();

    for &id in ids {
        let client = client.clone();
        let sem = Arc::clone(&sem);

        handles.push(tokio::spawn(async move {
            let _permit = sem.acquire().await.unwrap();
            client
                .get(format!("https://jsonplaceholder.typicode.com/posts/{id}"))
                .send()
                .await
                .ok()?
                .json::<serde_json::Value>()
                .await
                .ok()
        }));
    }

    let mut results = Vec::new();
    for h in handles {
        if let Ok(Some(v)) = h.await {
            results.push(v);
        }
    }
    results
}

#[tokio::main]
async fn main() -> Result<()> {
    let client = Client::new();
    let ids = [1, 2, 3, 4, 5];

    let posts = fetch_many(&client, &ids).await?;
    println!("fetched {} posts", posts.len());

    let posts = fetch_bounded(&client, &ids, 3).await; // max 3 in-flight
    println!("bounded: {} posts", posts.len());

    Ok(())
}
