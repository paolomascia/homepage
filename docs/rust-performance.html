Performance in Rust

Rust Performance Optimization

1. Measure Before Optimizing

Premature optimization is the root of all evil. Always profile first to find the actual bottleneck — intuition is often wrong.

# Build a release binary with debug symbols for profiling
# .cargo/config.toml:
# [profile.release]
# debug = true

cargo build --release

2. Allocation Strategies

Heap allocation is expensive. Common strategies to reduce it:

2.1 Pre-allocate with with_capacity

fn slow_collect(n: usize) -> Vec<u32> {
    let mut v = Vec::new(); // starts empty, reallocates multiple times
    for i in 0..n as u32 { v.push(i * i); }
    v
}

fn fast_collect(n: usize) -> Vec<u32> {
    let mut v = Vec::with_capacity(n); // one allocation, no reallocs
    for i in 0..n as u32 { v.push(i * i); }
    v
}

fn fastest_collect(n: usize) -> Vec<u32> {
    // Iterator collect already knows size via size_hint
    (0..n as u32).map(|i| i * i).collect()
}

fn main() {
    let _ = fast_collect(1_000_000);
    let _ = fastest_collect(1_000_000);
}

2.2 Reuse Buffers

// Instead of allocating a new buffer every call, reuse one
struct Processor {
    buffer: Vec<u8>,
}

impl Processor {
    fn new() -> Self { Processor { buffer: Vec::with_capacity(4096) } }

    fn process(&mut self, input: &[u8]) -> &[u8] {
        self.buffer.clear(); // reset length but keep capacity
        for &b in input {
            self.buffer.push(b.to_ascii_uppercase());
        }
        &self.buffer
    }
}

fn main() {
    let mut proc = Processor::new();
    for chunk in [b"hello".as_slice(), b"world"] {
        let result = proc.process(chunk);
        println!("{}", std::str::from_utf8(result).unwrap());
    }
    // Single allocation, reused for every call
}

use smallvec::{SmallVec, smallvec};

    let mut v: SmallVec<[u32; 8]> = smallvec![1, 2, 3, 4];
    v.push(5); // still on stack
    v.extend(6..14); // spills to heap only when needed

    println!("inline: {}", v.spilled()); // false while <= 8 elements

    // Cargo.toml: arrayvec = "0.7" — fixed max capacity, no heap fallback
}

3. Zero-Copy Parsing

Avoid allocating new strings when parsing — borrow slices of the input instead.

// Zero-copy CSV parser — borrows from the input string
#[derive(Debug)]
struct Record<'a> {
    name: &'a str,
    age: u32,
    city: &'a str,
}

fn parse_csv<'a>(input: &'a str) -> Vec<Record<'a>> {
    input.lines()
        .skip(1) // header
        .filter_map(|line| {
            let mut parts = line.splitn(3, ',');
            let name = parts.next()?.trim();
            let age  = parts.next()?.trim().parse().ok()?;
            let city = parts.next()?.trim();
            Some(Record { name, age, city })
        })
        .collect()
}

fn main() {
    let csv = "name,age,city\nAlice,30,Rome\nBob,25,Milan\nCarol,35,Turin";
    let records = parse_csv(csv); // zero heap allocations for strings!
    for r in &records {
        println!("{} ({}) from {}", r.name, r.age, r.city);
    }
}

3.1 nom for Zero-Copy Parsing

// Cargo.toml: nom = "7"
use nom::{
    bytes::complete::{tag, take_until},
    character::complete::{char, u32},
    sequence::separated_pair,
    IResult,
};

// Parse "key=value" pairs without allocation
fn parse_pair(input: &str) -> IResult<&str, (&str, &str)> {
    separated_pair(
        take_until("="),
        char('='),
        take_until("\n"),
    )(input)
}

fn main() {
    let input = "name=Alice\nage=30\n";
    match parse_pair(input) {
        Ok((_rest, (key, val))) => println!("{key} = {val}"),
        Err(e) => eprintln!("parse error: {e}"),
    }
}

4. Inlining and Monomorphization

// #[inline] hints that a function should be inlined at call sites
// Use sparingly — the compiler usually makes the right decision
#[inline]
fn add(a: i32, b: i32) -> i32 { a + b }

// #[inline(always)] forces inlining — use only for proven hot paths
#[inline(always)]
fn fast_min(a: f32, b: f32) -> f32 {
    if a < b { a } else { b }
}

// #[inline(never)] prevents inlining — useful for debugging or
// code size reduction when a large function is called in many places
#[inline(never)]
fn complex_error_handler(err: &str) {
    eprintln!("error: {err}");
}

// Monomorphization: generic functions get specialized copies per concrete type
// Faster than dynamic dispatch, but increases binary size
fn max_generic<T: PartialOrd>(a: T, b: T) -> T {
    if a > b { a } else { b }
}

// The compiler generates separate max_generic::<i32> and max_generic::<f64> versions
fn main() {
    println!("{}", max_generic(1i32, 2));
    println!("{}", max_generic(1.0f64, 2.0));
}

5. Cache Locality — Data-Oriented Design

Memory access patterns matter as much as algorithm complexity. Array-of-Structs (AoS) is intuitive; Struct-of-Arrays (SoA) is often faster for bulk processing because it maximizes cache use.

// Array-of-Structs (AoS) — intuitive, poor cache behavior for single-field access
struct ParticleAoS {
    x: f32, y: f32, z: f32,
    vx: f32, vy: f32, vz: f32,
    mass: f32,
}

fn update_positions_aos(particles: &mut [ParticleAoS], dt: f32) {
    for p in particles {
        // Accessing x, y, z: good. But we also drag mass, vx, vy, vz into cache.
        p.x += p.vx * dt;
        p.y += p.vy * dt;
        p.z += p.vz * dt;
    }
}

// Struct-of-Arrays (SoA) — cache-friendly: x array is contiguous
struct ParticlesSoA {
    x: Vec<f32>, y: Vec<f32>, z: Vec<f32>,
    vx: Vec<f32>, vy: Vec<f32>, vz: Vec<f32>,
    mass: Vec<f32>,
}

impl ParticlesSoA {
    fn update_positions(&mut self, dt: f32) {
        // Only x, vx touched here — both arrays fit in L1 cache together
        for i in 0..self.x.len() {
            self.x[i] += self.vx[i] * dt;
            self.y[i] += self.vy[i] * dt;
            self.z[i] += self.vz[i] * dt;
        }
    }

    // SIMD-friendly: compiler can auto-vectorize contiguous f32 arrays
    fn total_kinetic_energy(&self) -> f64 {
        self.vx.iter().zip(self.mass.iter())
            .zip(self.vy.iter())
            .zip(self.vz.iter())
            .map(|(((vx, m), vy), vz)| {
                let v2 = (vx * vx + vy * vy + vz * vz) as f64;
                0.5 * (*m as f64) * v2
            })
            .sum()
    }
}

fn main() {
    let n = 100_000;
    let mut particles = ParticlesSoA {
        x: vec![0.0f32; n], y: vec![0.0; n], z: vec![0.0; n],
        vx: (0..n).map(|i| i as f32 * 0.001).collect(),
        vy: vec![0.5f32; n], vz: vec![0.0; n],
        mass: vec![1.0f32; n],
    };

    particles.update_positions(0.016); // one frame at 60fps
    println!("KE = {:.2}", particles.total_kinetic_energy());
}

6. SIMD — Portable Simd

SIMD (Single Instruction, Multiple Data) processes multiple values in one CPU instruction. Rust's std::simd (nightly) provides a safe portable API; on stable, use wide or platform intrinsics.

// Auto-vectorization: write scalar, let the compiler vectorize
// Release mode with -C target-cpu=native enables AVX2 auto-vec
fn sum_floats(slice: &[f32]) -> f32 {
    // The compiler will typically auto-vectorize this with SIMD
    slice.iter().sum()
}

// Manual SIMD with the `wide` crate (stable)
// Cargo.toml: wide = "0.7"
use wide::f32x8;

fn dot_product_simd(a: &[f32], b: &[f32]) -> f32 {
    assert_eq!(a.len(), b.len());
    let chunks = a.len() / 8;

    let mut acc = f32x8::ZERO;
    for i in 0..chunks {
        let va = f32x8::from(&a[i * 8..i * 8 + 8]);
        let vb = f32x8::from(&b[i * 8..i * 8 + 8]);
        acc += va * vb;
    }

    // Sum the 8 lanes
    let mut result: f32 = acc.as_array_ref().iter().sum();

    // Handle remaining elements
    for i in (chunks * 8)..a.len() {
        result += a[i] * b[i];
    }

    result
}

fn main() {
    let a: Vec<f32> = (0..64).map(|x| x as f32).collect();
    let b: Vec<f32> = (0..64).map(|x| x as f32 * 2.0).collect();

    let scalar: f32 = a.iter().zip(b.iter()).map(|(x, y)| x * y).sum();
    let simd = dot_product_simd(&a, &b);

    println!("scalar: {scalar}, simd: {simd}");
    assert!((scalar - simd).abs() < 0.01);
}

6.1 Enabling Target Features

# Build for the current CPU with all available instructions
RUSTFLAGS="-C target-cpu=native" cargo build --release

# Build for a specific target CPU
RUSTFLAGS="-C target-feature=+avx2,+fma" cargo build --release

# Check what features the current CPU supports
rustc --print target-features | grep avx

// Runtime CPU feature detection — choose the best implementation at startup
fn process(data: &mut [f32]) {
    #[cfg(target_arch = "x86_64")]
    if is_x86_feature_detected!("avx2") {
        // Safety: we just verified AVX2 is available
        return unsafe { process_avx2(data) };
    }
    process_scalar(data)
}

fn process_scalar(data: &mut [f32]) {
    for x in data { *x *= 2.0; }
}

#[cfg(target_arch = "x86_64")]
#[target_feature(enable = "avx2")]
unsafe fn process_avx2(data: &mut [f32]) {
    // AVX2 intrinsics here — 8 floats at a time
    for chunk in data.chunks_exact_mut(8) {
        // use std::arch::x86_64::* intrinsics
        for x in chunk { *x *= 2.0; } // simplified; real code uses _mm256_*
    }
}

7. String and I/O Performance

use std::fmt::Write as FmtWrite;
use std::io::{self, Write, BufWriter};

// BAD: format! + push_str creates multiple intermediate allocations
fn build_string_slow(n: usize) -> String {
    let mut s = String::new();
    for i in 0..n {
        s += &format!("item-{i}\n"); // allocates a new String each iteration
    }
    s
}

// GOOD: write! into a String avoids intermediate allocations
fn build_string_fast(n: usize) -> String {
    let mut s = String::with_capacity(n * 10); // estimate capacity
    for i in 0..n {
        write!(&mut s, "item-{i}\n").unwrap(); // no allocation
    }
    s
}

// FAST stdout: use BufWriter to batch syscalls
fn print_many_lines(n: usize) {
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock()); // buffer writes
    for i in 0..n {
        writeln!(out, "line {i}").unwrap();
    }
    // BufWriter flushes on drop
}

fn main() {
    let s = build_string_fast(1000);
    println!("built {} chars", s.len());
    print_many_lines(100);
}

8. Lazy Initialization

use std::sync::OnceLock;

// Initialize once, read many times — thread-safe, zero overhead after init
static CONFIG: OnceLock<Config> = OnceLock::new();

#[derive(Debug)]
struct Config { db_url: String, pool_size: u32 }

fn get_config() -> &'static Config {
    CONFIG.get_or_init(|| {
        println!("initializing config (once)");
        Config {
            db_url: std::env::var("DATABASE_URL")
                .unwrap_or_else(|_| "postgres://localhost/mydb".to_string()),
            pool_size: 10,
        }
    })
}

// Lazy cell for non-Send types (single-threaded)
use std::cell::OnceCell;

struct Parser {
    regex_cache: OnceCell<String>, // expensive to compile
}

impl Parser {
    fn new() -> Self { Parser { regex_cache: OnceCell::new() } }

    fn get_pattern(&self) -> &str {
        self.regex_cache.get_or_init(|| {
            // expensive compilation happens only on first call
            r"\d{4}-\d{2}-\d{2}".to_string()
        })
    }
}

// Cargo.toml: once_cell = "1" (or use std::sync::LazyLock on Rust 1.80+)
// use once_cell::sync::Lazy;
// static COMPILED_REGEX: Lazy<regex::Regex> = Lazy::new(|| {
//     regex::Regex::new(r"\d{4}-\d{2}-\d{2}").unwrap()
// });

fn main() {
    println!("{:?}", get_config()); // initializing config (once)
    println!("{:?}", get_config()); // reuses existing
}

9. Optimizing Hot Loops

// Technique 1: bounds check elimination
fn sum_with_bounds_check(data: &[i32]) -> i64 {
    let mut total = 0i64;
    for i in 0..data.len() {
        total += data[i] as i64; // bounds check on each access
    }
    total
}

fn sum_no_bounds_check(data: &[i32]) -> i64 {
    // Iterator avoids bounds checks — compiler knows we're always in-bounds
    data.iter().map(|&x| x as i64).sum()
}

// Technique 2: chunk processing
fn process_chunks(data: &[u8]) -> u32 {
    let (chunks, remainder) = data.as_chunks::<16>(); // nightly
    let mut sum = 0u32;
    for chunk in chunks {
        for &b in chunk { sum += b as u32; }
    }
    for &b in remainder { sum += b as u32; }
    sum
}

// Technique 3: hint::unreachable_unchecked for impossible branches
fn classify(x: u8) -> &'static str {
    match x {
        0..=9   => "single digit",
        10..=99 => "double digit",
        _       => "triple digit",
    }
}

// Technique 4: use iterator adapters instead of explicit loops
fn efficient_pipeline(data: &[i32]) -> f64 {
    data.iter()
        .copied()
        .filter(|&x| x > 0)
        .map(|x| x as f64)
        .map(|x| x.sqrt())
        .sum::<f64>()
        / data.len() as f64
}

fn main() {
    let data: Vec<i32> = (1..=1000).collect();
    println!("{:.4}", efficient_pipeline(&data));
}

10. Performance Checklist

  1. Profile first — find the real bottleneck with a flamegraph or perf.
  2. Build in release mode — debug mode is 10-100x slower.
  3. Enable LTO and codegen-units=1 for production binaries.
  4. RUSTFLAGS="-C target-cpu=native" for local benchmarks.
  5. Pre-allocate with with_capacity when you know the size.
  6. Reuse buffers instead of allocating on every call.
  7. Prefer iterators over index loops — bounds checks are eliminated.
  8. Use &str and slices instead of String / Vec where possible.
  9. Check allocation counts with dhat — minimizing allocations often beats algorithmic improvements for hot paths.
  10. Benchmark with Criterion — wall clock time is noisy, use statistical benchmarks.
# Quick performance checklist commands
cargo build --release                         # always benchmark release
cargo flamegraph -- args                      # find hot functions
cargo bloat --release -n 20                   # find large functions
cargo bench                                   # run Criterion benchmarks
RUSTFLAGS="-C target-cpu=native" cargo bench  # with native CPU instructions
