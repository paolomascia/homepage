gRPC with Tonic

Rust gRPC Tonic

1. Setup and Protobuf Code Generation

# Cargo.toml
[dependencies]
tonic   = "0.12"
prost   = "0.13"
tokio   = { version = "1", features = ["full"] }
tokio-stream = "0.1"
futures = "0.3"

[build-dependencies]
tonic-build = "0.12"

// build.rs — generates Rust code from .proto files at compile time
fn main() -> Result<(), Box<dyn std::error::Error>> {
    tonic_build::configure()
        .build_server(true)
        .build_client(true)
        // Optional: output directory (default: OUT_DIR)
        // .out_dir("src/proto")
        .compile_protos(
            &["proto/user.proto", "proto/chat.proto"],
            &["proto/"],
        )?;
    Ok(())
}

// proto/user.proto
syntax = "proto3";
package user;

service UserService {
    // Unary: single request, single response
    rpc GetUser(GetUserRequest) returns (UserResponse);
    rpc CreateUser(CreateUserRequest) returns (UserResponse);
    rpc DeleteUser(DeleteUserRequest) returns (DeleteUserResponse);

    // Server-streaming: single request, stream of responses
    rpc ListUsers(ListUsersRequest) returns (stream UserResponse);

    // Client-streaming: stream of requests, single response
    rpc BatchCreateUsers(stream CreateUserRequest) returns (BatchCreateResponse);

    // Bidirectional streaming
    rpc Chat(stream ChatMessage) returns (stream ChatMessage);
}

message GetUserRequest     { string id = 1; }
message DeleteUserRequest  { string id = 1; }
message DeleteUserResponse { bool deleted = 1; }
message ListUsersRequest   { int32 page = 1; int32 per_page = 2; }
message BatchCreateResponse { int32 created = 1; repeated string ids = 2; }
message ChatMessage         { string sender = 1; string content = 2; int64 timestamp = 3; }

message CreateUserRequest {
    string name  = 1;
    string email = 2;
}

message UserResponse {
    string id         = 1;
    string name       = 2;
    string email      = 3;
    int64  created_at = 4;
}

2. Implementing the gRPC Server

// src/main.rs
use tonic::{transport::Server, Request, Response, Status};
use tokio_stream::wrappers::ReceiverStream;
use tokio::sync::mpsc;
use std::collections::HashMap;
use std::sync::{Arc, RwLock};

// Include the generated code
pub mod proto {
    tonic::include_proto!("user");
}

use proto::{
    user_service_server::{UserService, UserServiceServer},
    GetUserRequest, CreateUserRequest, DeleteUserRequest, DeleteUserResponse,
    ListUsersRequest, BatchCreateResponse, ChatMessage, UserResponse,
};

// Service implementation
#[derive(Clone, Default)]
struct UserServiceImpl {
    db: Arc<RwLock<HashMap<String, UserResponse>>>,
}

#[tonic::async_trait]
impl UserService for UserServiceImpl {
    // Unary RPC
    async fn get_user(
        &self,
        request: Request<GetUserRequest>,
    ) -> Result<Response<UserResponse>, Status> {
        let id = request.into_inner().id;
        let db = self.db.read().unwrap();

        db.get(&id)
            .cloned()
            .map(Response::new)
            .ok_or_else(|| Status::not_found(format!("user {id} not found")))
    }

    async fn create_user(
        &self,
        request: Request<CreateUserRequest>,
    ) -> Result<Response<UserResponse>, Status> {
        let req = request.into_inner();
        if req.name.is_empty() {
            return Err(Status::invalid_argument("name is required"));
        }

        let id = uuid_v4();
        let user = UserResponse {
            id: id.clone(),
            name: req.name,
            email: req.email,
            created_at: unix_now(),
        };

        self.db.write().unwrap().insert(id, user.clone());
        Ok(Response::new(user))
    }

    async fn delete_user(
        &self,
        request: Request<DeleteUserRequest>,
    ) -> Result<Response<DeleteUserResponse>, Status> {
        let deleted = self.db.write().unwrap()
            .remove(&request.into_inner().id)
            .is_some();
        Ok(Response::new(DeleteUserResponse { deleted }))
    }

    // Server-streaming RPC — return a stream
    type ListUsersStream = ReceiverStream<Result<UserResponse, Status>>;

    async fn list_users(
        &self,
        request: Request<ListUsersRequest>,
    ) -> Result<Response<Self::ListUsersStream>, Status> {
        let req = request.into_inner();
        let per_page = req.per_page.max(1) as usize;
        let offset   = (req.page.max(1) - 1) as usize * per_page;

        let users: Vec<UserResponse> = self.db.read().unwrap()
            .values()
            .skip(offset)
            .take(per_page)
            .cloned()
            .collect();

        let (tx, rx) = mpsc::channel(16);

        tokio::spawn(async move {
            for user in users {
                // Simulate some async work per item
                tokio::time::sleep(tokio::time::Duration::from_millis(10)).await;
                if tx.send(Ok(user)).await.is_err() {
                    break; // client disconnected
                }
            }
        });

        Ok(Response::new(ReceiverStream::new(rx)))
    }

    // Client-streaming RPC — receive a stream, return a single response
    async fn batch_create_users(
        &self,
        request: Request<tonic::Streaming<CreateUserRequest>>,
    ) -> Result<Response<BatchCreateResponse>, Status> {
        let mut stream = request.into_inner();
        let mut ids = Vec::new();

        while let Some(req) = stream.message().await? {
            if req.name.is_empty() {
                return Err(Status::invalid_argument("all users must have a name"));
            }
            let id = uuid_v4();
            let user = UserResponse {
                id: id.clone(),
                name: req.name,
                email: req.email,
                created_at: unix_now(),
            };
            self.db.write().unwrap().insert(id.clone(), user);
            ids.push(id);
        }

        Ok(Response::new(BatchCreateResponse {
            created: ids.len() as i32,
            ids,
        }))
    }

    // Bidirectional streaming
    type ChatStream = ReceiverStream<Result<ChatMessage, Status>>;

    async fn chat(
        &self,
        request: Request<tonic::Streaming<ChatMessage>>,
    ) -> Result<Response<Self::ChatStream>, Status> {
        let mut stream = request.into_inner();
        let (tx, rx) = mpsc::channel(16);

        tokio::spawn(async move {
            while let Some(msg) = stream.message().await.unwrap_or(None) {
                let reply = ChatMessage {
                    sender: "server".to_string(),
                    content: format!("echo: {}", msg.content),
                    timestamp: unix_now(),
                };
                if tx.send(Ok(reply)).await.is_err() { break; }
            }
        });

        Ok(Response::new(ReceiverStream::new(rx)))
    }
}

fn uuid_v4() -> String {
    format!("{:x}", std::time::SystemTime::now()
        .duration_since(std::time::UNIX_EPOCH).unwrap().as_nanos())
}
fn unix_now() -> i64 {
    std::time::SystemTime::now()
        .duration_since(std::time::UNIX_EPOCH).unwrap().as_secs() as i64
}

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let addr = "0.0.0.0:50051".parse()?;
    let svc = UserServiceImpl::default();

    println!("gRPC server listening on {addr}");

    Server::builder()
        .add_service(UserServiceServer::new(svc))
        .serve(addr)
        .await?;

    Ok(())
}

3. gRPC Client

use tonic::transport::Channel;
use tokio_stream::StreamExt;

pub mod proto {
    tonic::include_proto!("user");
}

use proto::{
    user_service_client::UserServiceClient,
    CreateUserRequest, GetUserRequest, ListUsersRequest, ChatMessage,
};

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    // Connect to the server
    let channel = Channel::from_static("http://localhost:50051")
        .connect()
        .await?;

    let mut client = UserServiceClient::new(channel);

    // Unary call
    let response = client.create_user(CreateUserRequest {
        name: "Alice".to_string(),
        email: "alice@example.com".to_string(),
    }).await?;
    let user = response.into_inner();
    println!("created: {:?}", user);

    // Unary call — get by ID
    let response = client.get_user(GetUserRequest {
        id: user.id.clone(),
    }).await?;
    println!("fetched: {:?}", response.into_inner());

    // Server-streaming call
    let mut stream = client.list_users(ListUsersRequest {
        page: 1, per_page: 10,
    }).await?.into_inner();

    println!("listing users:");
    while let Some(user) = stream.next().await {
        println!("  {:?}", user?);
    }

    // Client-streaming call
    let requests = vec![
        CreateUserRequest { name: "Bob".to_string(),   email: "bob@example.com".to_string() },
        CreateUserRequest { name: "Carol".to_string(), email: "carol@example.com".to_string() },
    ];
    let response = client.batch_create_users(
        tokio_stream::iter(requests)
    ).await?;
    println!("batch created: {:?}", response.into_inner());

    // Bidirectional streaming
    let messages = vec![
        ChatMessage { sender: "client".to_string(), content: "hello".to_string(), timestamp: 0 },
        ChatMessage { sender: "client".to_string(), content: "world".to_string(), timestamp: 0 },
    ];

    let mut bidi = client.chat(tokio_stream::iter(messages)).await?.into_inner();
    while let Some(reply) = bidi.next().await {
        println!("server says: {:?}", reply?);
    }

    Ok(())
}

4. Interceptors — Metadata and Auth

use tonic::{Request, Status, service::Interceptor};

// Server-side interceptor: validate API key in metadata
fn auth_interceptor(req: Request<()>) -> Result<Request<()>, Status> {
    let token = req.metadata().get("x-api-key")
        .and_then(|v| v.to_str().ok());

    match token {
        Some(t) if t == "valid-key" => Ok(req),
        Some(_) => Err(Status::unauthenticated("invalid API key")),
        None    => Err(Status::unauthenticated("missing x-api-key header")),
    }
}

// Apply interceptor to the server
Server::builder()
    .add_service(
        UserServiceServer::with_interceptor(UserServiceImpl::default(), auth_interceptor)
    )
    .serve(addr)
    .await?;

// Client-side interceptor: attach auth token to every request
use tonic::codegen::InterceptedService;

fn client_auth_interceptor(mut req: Request<()>) -> Result<Request<()>, Status> {
    req.metadata_mut().insert(
        "x-api-key",
        "valid-key".parse().unwrap(),
    );
    Ok(req)
}

let client = UserServiceClient::with_interceptor(channel, client_auth_interceptor);

5. Error Handling and Status Codes

use tonic::Status;

// Map domain errors to gRPC status codes
fn to_status(err: &str) -> Status {
    if err.contains("not found")    { return Status::not_found(err); }
    if err.contains("unauthorized") { return Status::unauthenticated(err); }
    if err.contains("forbidden")    { return Status::permission_denied(err); }
    if err.contains("conflict")     { return Status::already_exists(err); }
    if err.contains("invalid")      { return Status::invalid_argument(err); }
    if err.contains("timeout")      { return Status::deadline_exceeded(err); }
    Status::internal(err)
}

// Rich error details using tonic-types
// Cargo.toml: tonic-types = "0.12"
use tonic_types::{ErrorDetails, StatusExt};

fn validation_error(field: &str, description: &str) -> Status {
    let mut details = ErrorDetails::new();
    details.add_bad_request_violation(field, description);
    Status::with_error_details(
        tonic::Code::InvalidArgument,
        "validation failed",
        details,
    ).unwrap_or_else(|_| Status::invalid_argument("validation failed"))
}

6. TLS Configuration

use tonic::transport::{Server, ServerTlsConfig, Identity, ClientTlsConfig, Certificate};

// Server with TLS
async fn tls_server() -> Result<(), Box<dyn std::error::Error>> {
    let cert = tokio::fs::read("certs/server.crt").await?;
    let key  = tokio::fs::read("certs/server.key").await?;
    let identity = Identity::from_pem(cert, key);

    Server::builder()
        .tls_config(ServerTlsConfig::new().identity(identity))?
        .add_service(UserServiceServer::new(UserServiceImpl::default()))
        .serve("0.0.0.0:50051".parse()?)
        .await?;

    Ok(())
}

// Client with TLS
async fn tls_client() -> Result<(), Box<dyn std::error::Error>> {
    let ca_cert = tokio::fs::read("certs/ca.crt").await?;
    let ca_cert = Certificate::from_pem(ca_cert);

    let tls = ClientTlsConfig::new()
        .ca_certificate(ca_cert)
        .domain_name("localhost");

    let channel = Channel::from_static("https://localhost:50051")
        .tls_config(tls)?
        .connect()
        .await?;

    let _client = UserServiceClient::new(channel);
    Ok(())
}

7. Reflection and grpcurl Testing

# Cargo.toml
[dependencies]
tonic-reflection = "0.12"

use tonic_reflection::server::Builder as ReflectionBuilder;

// Register proto descriptors for server reflection
// This allows tools like grpcurl to discover your API dynamically
const FILE_DESCRIPTOR_SET: &[u8] =
    tonic::include_file_descriptor_set!("user_descriptor");

Server::builder()
    .add_service(
        ReflectionBuilder::configure()
            .register_encoded_file_descriptor_set(FILE_DESCRIPTOR_SET)
            .build_v1()
            .unwrap()
    )
    .add_service(UserServiceServer::new(UserServiceImpl::default()))
    .serve(addr)
    .await?;

8. gRPC + REST Gateway

Expose gRPC services as a REST API alongside the native gRPC endpoint using Axum as a gateway:

// Run both gRPC and HTTP on the same port using tower
use axum::{routing::post, Router, Json, extract::Path};

async fn rest_create_user(Json(body): Json<serde_json::Value>) -> impl axum::response::IntoResponse {
    // Call the gRPC service internally
    let channel = Channel::from_static("http://localhost:50051").connect().await.unwrap();
    let mut client = UserServiceClient::new(channel);

    let response = client.create_user(CreateUserRequest {
        name: body["name"].as_str().unwrap_or("").to_string(),
        email: body["email"].as_str().unwrap_or("").to_string(),
    }).await.unwrap();

    Json(serde_json::json!({
        "id": response.into_inner().id
    }))
}

// Or use tonic-web to serve gRPC-Web (browser-compatible gRPC)
// Cargo.toml: tonic-web = "0.12"
Server::builder()
    .accept_http1(true)                    // required for gRPC-Web
    .layer(tonic_web::GrpcWebLayer::new()) // wraps gRPC for browser clients
    .add_service(UserServiceServer::new(UserServiceImpl::default()))
    .serve(addr)
    .await?;
